完整實作// Full C++

p473.cpp C++17 
#include <bits/stdc++.h>
using namespace std;

// ── 並查集（路徑壓縮 + 按秩合併）
int par[100005], rnk[100005];

int find(int x) {
    while (par[x] != x) { par[x] = par[par[x]]; x = par[x]; }
    return x;
}

bool unite(int a, int b) {
    a = find(a); b = find(b);
    if (a == b) return false;
    if (rnk[a] < rnk[b]) swap(a, b);
    par[b] = a;
    if (rnk[a] == rnk[b]) rnk[a]++;
    return true;
}

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    vector<array<long long, 3>> es(M);   // {w, u, v}，sort 先比 w
    for (int i = 0; i < M; i++) {
        long long u, v, w;
        cin >> u >> v >> w;
        es[i] = {w, u, v};
    }
    sort(es.begin(), es.end());

    for (int i = 1; i <= N; i++) { par[i] = i; rnk[i] = 0; }

    long long deletions = 0;
    int i = 0;
    while (i < M) {
        int j = i;
        long long w = es[i][0];

        // 1) 收集此權重類別「連接不同連通塊」的候選邊
        //    （先收集再合併，避免同組邊互相改變 DSU 狀態）
        vector<pair<int, int>> cand;
        while (j < M && es[j][0] == w) {
            int u = es[j][1], v = es[j][2];
            if (find(u) != find(v)) cand.push_back({u, v});
            j++;
        }

        // 2) 對候選邊合併，數真正造成合併的次數
        int merges = 0;
        for (auto [u, v] : cand)
            if (unite(u, v)) merges++;

        // 3) 候選數 − 合併數 = 此類別冗餘邊（製造非唯一的選擇）
        deletions += (long long)cand.size() - merges;

        i = j;
    }

    cout << deletions << '\n';
}

int main() { solve(); return 0; }

「先收集候選、再合併」很重要：同一權重類別內的邊不能邊掃邊 union， 否則前面的 union 會改變後面 find 的結果，把同組互相平行的冗餘邊誤判掉。

03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
先收集再合併 同一權重組必須先把所有候選邊收集完（用組前的 DSU 判 find），再統一 union。邊掃邊 union 會讓組內互相平行的冗餘邊被前一條提早合併、漏算刪除數。
候選 = 不同塊 只有 find(u) ≠ find(v) 的邊才算候選。兩端早已同塊（被更輕的邊連起）的邊不影響 MST，不算進冗餘。
冗餘 = 候選 − 合併 一組 c 條候選只需 merges 次合併 ⇒ 多出 c − merges 條是選擇來源，全刪。
不改最小花費 刪的都是「同權重多餘候選」，不影響每個權重的合併次數 ⇒ 最小花費保證不變。
權重相同才同組 分組依據是權重值相等，不是排序後相鄰。用 while (es[j][0]==w) 切組。
I/O 速度 M 達 3×10⁵ ⇒ sync_with_stdio(false); cin.tie(nullptr); 必加。

細節走讀// Section Walkthrough

① 讀邊 + 排序

// read & sort
es[i] = {w, u, v}          ← w 放最前，sort 先比權重
sort(es)                   ← 由小到大

② 切權重類別

// group by weight
i = 0
while i < M:
    j = i; w = es[i].w
    while j < M and es[j].w == w: j++   ← [i, j) 同權重

③ 收集候選（用組前 DSU 判定）

// collect candidates (關鍵)
cand = []
for e in [i, j):
    if find(u) != find(v):     ← 此刻 DSU 還沒被本組合併過
        cand.push_back((u, v))

④ 合併 + 數冗餘

// merge & count
merges = 0
for (u, v) in cand:
    if unite(u, v): merges++    ← 真正造成合併
deletions += |cand| - merges    ← 冗餘 = 候選 − 合併
i = j
