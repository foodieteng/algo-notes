01

完整實作// Full C++

p115.cpp C++17 
#include <bits/stdc++.h>
using namespace std;

const long long MOD = 10000019;
const int       N_MAX = 1000005;

long long a[N_MAX];      // 原序列 (排序時就地更新)
long long tmp[N_MAX];    // merge 用的暫存陣列
long long inv_sum = 0;  // 累積的逆序對「兩數和」(已 mod)

// 合併 a[left..mid] 與 a[mid+1..right]，
// 同時把跨界逆序對的兩數和累進 inv_sum。
void merge_range(int left, int mid, int right) {
    // remaining = a[left..mid] 之和 (mod MOD)，左半游標前進時遞減
    long long remaining = 0;
    for (int p = left; p <= mid; ++p)
        remaining = (remaining + a[p] % MOD) % MOD;

    int i = left;          // 左半游標
    int j = mid + 1;       // 右半游標
    int k = left;          // 寫入 tmp 的位置

    while (i <= mid && j <= right) {
        if (a[i] <= a[j]) {
            // 取左半：無新逆序對；remaining 扣掉 a[i]
            tmp[k++] = a[i];
            remaining = (remaining - a[i] % MOD + MOD) % MOD;
            ++i;
        } else {
            // a[i] > a[j]：左半剩 count 個全 > a[j]，全部跟它配成逆序對
            long long count   = mid - i + 1;
            long long contrib = (remaining + (a[j] % MOD) * (count % MOD)) % MOD;
            inv_sum = (inv_sum + contrib) % MOD;
            tmp[k++] = a[j];
            ++j;
        }
    }
    while (i <= mid)   tmp[k++] = a[i++];
    while (j <= right) tmp[k++] = a[j++];

    for (int p = left; p <= right; ++p) a[p] = tmp[p];
}

void merge_sort(int left, int right) {
    if (left >= right) return;
    int mid = (left + right) / 2;
    merge_sort(left,     mid);
    merge_sort(mid + 1, right);
    merge_range(left, mid, right);
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);

    int n; cin >> n;
    for (int i = 0; i < n; ++i) cin >> a[i];

    merge_sort(0, n - 1);

    cout << inv_sum << '\n';
    return 0;
}

複雜度分析// Complexity Derivation

把 merge_sort(left, right) 一層的工作量分成三段：

步驟內容單層成本
遞迴左右兩個 N/2 子問題2 · T(N/2)
算 remaining掃左半 N/2 個元素一次求和O(N)
雙指針 mergei、j 各走過所有元素一遍O(N)
跨界貢獻累加每次取右半 O(1)，最多取 N 次O(N)
T(N) = 2 · T(N/2) + O(N)
Master Theorem case 2：a = 2, b = 2, f(N) = O(N) = O(Nlogb a) ⇒ T(N) = Θ(N log N)
項目數量備註
TimeO(N log N)N = 10⁶ ⇒ 約 2 × 10⁷ 次運算
SpaceO(N)全域 tmp[] 暫存；遞迴堆疊 O(log N)
Recursion DepthO(log N)N = 10⁶ ⇒ 約 20 層
Constant factor低純整數比較、無浮點、無 sqrt

與單純的 計逆序對數量 比起來，多算的只是「每次取右時加一筆 remaining + a[j]×count」— 仍是 O(1)，總複雜度不變。

03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
比較永遠用原始值 mod 會破壞順序 — 不同的大數可能對映到同一餘數。a[i] <= a[j] 必須用未 mod 的原始值。
全程 long long (a[j] % MOD) × (count % MOD) ≤ 10⁷ × 10⁶ = 10¹³ — 遠超 int 上限 (2.1 × 10⁹)。remaining、contrib、inv_sum、陣列 a/tmp 都用 long long。
減法要 + MOD (remaining - a[i] % MOD + MOD) % MOD — 不加 MOD 結果會變負，導致下一步累加錯誤。
tmp 用全域 10⁶ × log(10⁶) ≈ 2 × 10⁷ 次 malloc/free 開銷太大。static long long tmp[N_MAX] 一次配置永久使用。
copy back 別忘 merge 完後 for (p = left..right) a[p] = tmp[p] — 不寫回的話下一層 merge 還是讀到舊資料。
剩餘單邊不算貢獻 主迴圈結束時，若左半還有剩，那些元素全部小於右半被吃光的所有元素 — 不會再產生新跨界逆序對，不要更新 inv_sum。
I/O 速度 ios::sync_with_stdio(false); cin.tie(nullptr); 必加；10⁶ 筆輸入用 cin/cout 沒這兩行會 TLE。輸出用 '\n'，別用 endl。
04

細節走讀// Section Walkthrough

① 切半遞迴

// split + recurse
if (left >= right) return;          ← base：單一元素，無逆序對
mid = (left + right) / 2;
merge_sort(left,     mid);          ← 左半排好、左半內部逆序對計完
merge_sort(mid + 1, right);         ← 右半排好、右半內部逆序對計完
merge_range(left, mid, right);      ← 兩半合併 + 算「跨界」逆序對

② 初始化 remaining (左半剩餘和)

// remaining init
remaining = 0
for p in [left, mid]:
    remaining = (remaining + a[p] % MOD) % MOD

意義：左半當前剩餘元素之和 (mod MOD)。
之後每從左半取出一個 a[i]，就 remaining -= a[i]。
這樣 contrib 計算可以 O(1) 取得「左半剩餘和」。

③ 雙指針 merge — 取左 vs 取右

// two-pointer merge
i = left, j = mid+1, k = left
while i <= mid AND j <= right:
    if a[i] <= a[j]:               ← 比較用原始值！
        tmp[k++] = a[i]
        remaining -= a[i] (mod)    ← 別忘了扣
        ++i
    else:                          ← a[i] > a[j]，是逆序對群
        count   = mid - i + 1       ← 左半剩餘個數
        contrib = remaining + a[j] × count   ← 一次累積
        inv_sum += contrib (mod)
        tmp[k++] = a[j]
        ++j

關鍵公式：contrib = remaining + a[j] × count — 左半剩 count 個元素，每個跟 a[j] 配成逆序對，那些元素的和就是 remaining， a[j] 出現 count 次貢獻 a[j] × count。

④ 收尾 + 寫回原陣列

// drain + copy back
while i <= mid:   tmp[k++] = a[i++]   ← 左半剩餘倒進 tmp
while j <= right: tmp[k++] = a[j++]   ← 右半剩餘倒進 tmp

for p in [left, right]:
    a[p] = tmp[p]                      ← 寫回，上層 merge 才能讀到

為什麼收尾不算貢獻？若主迴圈結束時左半還有剩，代表右半被吃完了。那些右半元素都已比左半剩的還小（不然不會被先取出）， 所以左半剩的元素跟已吃光的右半沒有逆序對關係 — 在前面 if 分支已經處理過了。
