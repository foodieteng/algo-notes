完整實作// Full C++

p500.cpp C++17 
#include <bits/stdc++.h>
using namespace std;

const int N_MAX = 500005;

int N;
int a[N_MAX];
int maxL[N_MAX], minL[N_MAX];   // 由 mid 往左累積
int maxR[N_MAX], minR[N_MAX];   // 由 mid+1 往右累積

// 計 cases 3/4 的等式用 count[]，index 範圍 [0, 2N+5]。
// case 3: index = R + minR[R]    ∈ [2, 2N]
// case 4: index = maxR[R] − R + N（shifted），原值範圍 [−(N−1), N−1]
int cnt[2 * N_MAX + 10];

// 計算跨界 (L ≤ mid < R) 的 good (L, R) 數量
long long count_crossing(int lo, int mid, int hi) {
    // --- prefix max/min ---
    maxL[mid] = minL[mid] = a[mid];
    for (int L = mid - 1; L >= lo; --L) {
        maxL[L] = max(maxL[L+1], a[L]);
        minL[L] = min(minL[L+1], a[L]);
    }
    maxR[mid+1] = minR[mid+1] = a[mid+1];
    for (int R = mid + 2; R <= hi; ++R) {
        maxR[R] = max(maxR[R-1], a[R]);
        minR[R] = min(minR[R-1], a[R]);
    }

    long long ans = 0;

    // === CASE 1 · max + min 都在左 ===
    // good ⟺ R = L + maxL[L] − minL[L]，且 R 落在 (mid, hi]，
    // 且右側 max/min 不破壞：maxR[R] < maxL[L]、minR[R] > minL[L]
    for (int L = lo; L <= mid; ++L) {
        int R = L + maxL[L] - minL[L];
        if (R > mid && R <= hi &&
            maxR[R] < maxL[L] && minR[R] > minL[L]) {
            ++ans;
        }
    }

    // === CASE 2 · max + min 都在右 ===
    // good ⟺ L = R − (maxR[R] − minR[R])，L 必須落在 [lo, mid]
    for (int R = mid + 1; R <= hi; ++R) {
        int L = R - (maxR[R] - minR[R]);
        if (L >= lo && L <= mid &&
            maxL[L] < maxR[R] && minL[L] > minR[R]) {
            ++ans;
        }
    }

    // === CASE 3 · max 在左、min 在右 ===
    // 條件：maxL[L] > maxR[R]、minR[R] < minL[L]
    // good ⟺ L + maxL[L] == R + minR[R]
    // 兩個指針 pmax/pmin 都單調右移 (L 由 mid 往下時 maxL[L] 不減、minL[L] 不增)
    {
        int pmax = mid;      // 已加入 (max-side qualifies)：(mid, pmax]
        int pmin = mid + 1;  // 仍不滿足 min-side 的 R：(mid, pmin); 滿足者：[pmin, hi]
        for (int L = mid; L >= lo; --L) {
            // 推進 pmax：maxR[pmax+1] < maxL[L] 時，pmax++，若該 R 也滿足 min-side 則加入 cnt
            while (pmax + 1 <= hi && maxR[pmax + 1] < maxL[L]) {
                ++pmax;
                if (pmax >= pmin) ++cnt[pmax + minR[pmax]];
            }
            // 推進 pmin：minR[pmin] >= minL[L] 表示 R=pmin 不滿足 min-side，若之前在 cnt 內就移除
            while (pmin <= hi && minR[pmin] >= minL[L]) {
                if (pmin <= pmax) --cnt[pmin + minR[pmin]];
                ++pmin;
            }
            ans += cnt[L + maxL[L]];
        }
        // 清掉這次留下的 cnt 痕跡 (active set 是 [pmin, pmax])
        for (int R = pmin; R <= pmax; ++R) cnt[R + minR[R]] = 0;
    }

    // === CASE 4 · max 在右、min 在左 ===
    // 條件：maxR[R] > maxL[L]、minR[R] > minL[L]
    // good ⟺ maxR[R] − R == minL[L] − L
    // cnt index 用 (maxR[R] − R + N)，避免負數
    {
        int qlo = mid + 1;  // max-side 滿足者：[qlo, hi]；qlo 由 mid+1 往右單調推進
        int qhi = mid;      // min-side 滿足者：(mid, qhi]；qhi 由 mid 往右單調推進
        for (int L = mid; L >= lo; --L) {
            // 推進 qhi：minR[qhi+1] > minL[L] 時，加入 R=qhi+1 (若 max-side 也滿足)
            while (qhi + 1 <= hi && minR[qhi + 1] > minL[L]) {
                ++qhi;
                if (qhi >= qlo) ++cnt[maxR[qhi] - qhi + N];
            }
            // 推進 qlo：maxR[qlo] <= maxL[L] 表示 R=qlo 不滿足 max-side，移除
            while (qlo <= hi && maxR[qlo] <= maxL[L]) {
                if (qlo <= qhi) --cnt[maxR[qlo] - qlo + N];
                ++qlo;
            }
            ans += cnt[minL[L] - L + N];
        }
        // 清掉 cnt 殘留
        for (int R = qlo; R <= qhi; ++R) cnt[maxR[R] - R + N] = 0;
    }

    return ans;
}

long long solve(int lo, int hi) {
    if (lo == hi) return 1;           // 單一元素 = 1 個 good
    int mid = (lo + hi) / 2;
    return solve(lo, mid)
         + solve(mid + 1, hi)
         + count_crossing(lo, mid, hi);
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);

    int T; cin >> T;
    while (T--) {
        cin >> N;
        for (int i = 1; i <= N; ++i) cin >> a[i];
        cout << solve(1, N) << '\n';
    }
    return 0;
}

複雜度分析// Complexity Derivation

把 count_crossing(lo, mid, hi) 拆成 5 段（設 K = hi − lo + 1）：

步驟內容單次成本
建 maxL/minL/maxR/minR掃過左右兩半各一次O(K)
Case 1每個 L 一次 O(1) 檢查O(K)
Case 2每個 R 一次 O(1) 檢查O(K)
Case 3 (two-pointer)L 走一遍 + pmax/pmin 各走一遍O(K)
Case 4 (two-pointer)L 走一遍 + qlo/qhi 各走一遍O(K)
cnt 清理active set 大小 ≤ K，O(K) 歸零O(K)
T(N) = 2 · T(N/2) + O(N)
Master Theorem case 2：a = 2, b = 2, f(N) = O(N) = O(Nlogb a) ⇒ T(N) = Θ(N log N)
項目數量備註
Time (單組)O(N log N)N = 5 × 10⁵ ⇒ 約 10⁷ 次運算
Time (全部 T 組)O(S log S)S ≤ 5 × 10⁵ ⇒ ~10⁷ 級，安全過時限
SpaceO(N)a, maxL, minL, maxR, minR 各一支 + cnt 約 2N
Recursion DepthO(log N)N = 5 × 10⁵ ⇒ 約 20 層
Constant factor中四個 case + 兩個 two-pointer + cnt 加減

邊界與陷阱// Pitfalls

陷阱說明 / 解法
答案型別 long long N = 5 × 10⁵ 全為「已排序」時答案 = N(N+1)/2 ≈ 1.25 × 10¹¹，遠超 int 上限（2.1 × 10⁹）。 solve、count_crossing 都回傳 long long。
跨 test case 殘留 cnt[] 在每個 case 3/4 結束都必須歸零；本實作在 count_crossing 結尾局部清掉 active set 範圍內的 index，避免後續 case 看到髒資料。
cnt 陣列範圍 case 3 的 index 是 R + minR[R] ∈ [2, 2N]； case 4 的 index 加了 + N 偏移後 ∈ [1, 2N − 1]。 開到 2 · N_MAX + 10 才安全。
1-indexed vs 0-indexed 實作採 1-indexed（a[1..N]）。lo == hi 即 base case；mid = (lo + hi)/2； crossing 範圍 L ∈ [lo, mid]、R ∈ [mid+1, hi]。
嚴格 vs 非嚴格 permutation ⇒ 元素皆不同，所以 maxL[L] ≠ maxR[R]（不同位置的 max 不可能相等）， 可放心用 嚴格不等 > / <。
two-pointer 不要 reset case 3 內 pmax/pmin 全程單調右移，整個 L 迴圈共用一對指針，不在每個 L 重新初始化。 這是達到 O(N) 而非 O(N²) 的關鍵。
IO 速度 T、N 都到 5 × 10⁵ 級，必加 ios::sync_with_stdio(false); cin.tie(nullptr); 或改用 scanf/printf。 輸出用 '\n' 別用 endl。

細節走讀// 4-Case Walkthrough

① CASE 1：max + min 都在左半

// CASE 1: max+min on LEFT
條件：max = maxL[L]、min = minL[L]
goodness：maxL[L] − minL[L] = R − L
       ⇒ R = L + maxL[L] − minL[L]   ← 對每個 L 直接算出唯一候選 R

驗證右側不破壞：maxR[R] < maxL[L]、minR[R] > minL[L]
   （若右側出現更大值或更小值，max/min 就不是「來自左半」了）

for L in [lo, mid]:
    R = L + maxL[L] − minL[L]
    if R in (mid, hi] AND maxR[R] < maxL[L] AND minR[R] > minL[L]:
        ans += 1

② CASE 2：max + min 都在右半 · 對偶

// CASE 2: max+min on RIGHT
條件：max = maxR[R]、min = minR[R]
goodness：maxR[R] − minR[R] = R − L
       ⇒ L = R − (maxR[R] − minR[R])   ← 對每個 R 直接算出唯一候選 L

驗證左側不破壞：maxL[L] < maxR[R]、minL[L] > minR[R]

for R in (mid, hi]:
    L = R − (maxR[R] − minR[R])
    if L in [lo, mid] AND maxL[L] < maxR[R] AND minL[L] > minR[R]:
        ans += 1

③ CASE 3：max 在左、min 在右 · two-pointer 計數

// CASE 3: max LEFT, min RIGHT
條件：maxL[L] > maxR[R] AND minR[R] < minL[L]
goodness：maxL[L] − minR[R] = R − L
       ⇒ L + maxL[L] = R + minR[R]   ← 等式可用 cnt[] hashing

L 由 mid 向 lo 走：maxL[L] 不減、minL[L] 不增
pmax (max-side 滿足者上界) ↗ 只右移：maxL 變大 ⇒ 更多 R 滿足
pmin (min-side 不滿足者上界+1) ↗ 只右移：minL 變小 ⇒ 更少 R 滿足
active set = R ∈ [pmin, pmax]

每加入 R ⇒ cnt[R + minR[R]]++
每移除 R ⇒ cnt[R + minR[R]]−−
查詢      ⇒ ans += cnt[L + maxL[L]]

④ CASE 4：max 在右、min 在左 · 對偶 two-pointer

// CASE 4: max RIGHT, min LEFT
條件：maxR[R] > maxL[L] AND minR[R] > minL[L]
goodness：maxR[R] − minL[L] = R − L
       ⇒ maxR[R] − R = minL[L] − L   ← 兩邊都跟「位置」綁在一起

cnt 用 (maxR[R] − R + N) 當 index（避免負數）

L 由 mid 向 lo 走：maxL[L] 不減 ⇒ qlo 右移（max-side 失去 R）
              minL[L] 不增 ⇒ qhi 右移（min-side 多收 R）
active set = R ∈ [qlo, qhi]

查詢 ⇒ ans += cnt[minL[L] − L + N]

為什麼是 O(N log N)？case 3/4 的 two-pointer 在 L 迴圈結束時，pmax/qhi 最多走過 hi−mid 步、 pmin/qlo 也最多走過 hi−mid 步 ⇒ 整個 case 在 O(hi − lo + 1) 內完成。每層遞迴掃 O(N)，共 log N 層。
