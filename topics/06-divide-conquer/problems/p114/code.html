p114.cpp C++17 
#include <array>
#include <iostream>
#include "lib0114.h"

using namespace std;

struct Cell {
    int x, y;
};

void report(const array<Cell, 3>& cells) {
    Report(cells[0].x, cells[0].y,
           cells[1].x, cells[1].y,
           cells[2].x, cells[2].y);
}

// 鋪滿左上角為 (top, left)、邊長為 n 的子棋盤。
// blocked 代表這個子棋盤中已經不能再使用的格子。
void tile(int top, int left, int n, Cell blocked) {
    if (n == 2) {
        array<Cell, 3> cells{};
        int count = 0;

        for (int dx = 0; dx < 2; ++dx) {
            for (int dy = 0; dy < 2; ++dy) {
                Cell cur{top + dx, left + dy};
                if (cur.x == blocked.x && cur.y == blocked.y) continue;
                cells[count++] = cur;
            }
        }

        report(cells);
        return;
    }

    const int half = n / 2;
    const int midX = top + half - 1;
    const int midY = left + half - 1;

    // TL, TR, BL, BR 四個象限最靠近中心的格子。
    array<Cell, 4> center = {{
        {midX,     midY},
        {midX,     midY + 1},
        {midX + 1, midY},
        {midX + 1, midY + 1}
    }};

    // blocked 落在哪個象限？0=TL, 1=TR, 2=BL, 3=BR。
    const int blockedQuadrant = (blocked.x > midX) * 2 + (blocked.y > midY);

    // 中央 L 放在「沒有原 blocked」的三個象限。
    array<Cell, 3> middleL{};
    int count = 0;
    for (int q = 0; q < 4; ++q) {
        if (q == blockedQuadrant) continue;
        middleL[count++] = center[q];
    }
    report(middleL);

    array<Cell, 4> origin = {{
        {top,        left},
        {top,        left + half},
        {top + half, left},
        {top + half, left + half}
    }};

    // 原 blocked 所在象限沿用它；其他象限把中央 L 的那格當成 blocked。
    for (int q = 0; q < 4; ++q) {
        Cell nextBlocked = (q == blockedQuadrant) ? blocked : center[q];
        tile(origin[q].x, origin[q].y, half, nextBlocked);
    }
}

void solve(int N, int X, int Y) {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    tile(1, 1, N, Cell{X, Y});
}

逐段走讀

核心是 tile(top, left, n, blocked)：對一個左上角 (top,left)、邊長 n、已有一格不可使用的子板鋪 L。

if (n == 2) {
    /* 4 格扣掉 blocked 後 report */
}
Base case。 2×2 板只有 4 格，扣掉 blocked 後剩 3 格必然構成 L。 直接掃 4 格，跳過 blocked，剩 3 格送進 report()。 
const int half = n / 2;
const int midX = top + half - 1;
const int midY = left + half - 1;

array<Cell, 4> center = {{
    {midX,     midY},
    {midX,     midY + 1},
    {midX + 1, midY},
    {midX + 1, midY + 1}
}};
中央 4 角座標。 每個象限「最靠近中心」的格子，按 TL / TR / BL / BR 順序排： (midX,midY)、(midX,midY+1)、(midX+1,midY)、(midX+1,midY+1)。 這 4 格就是放置中央 L 的候選。 
const int blockedQuadrant =
    (blocked.x > midX) * 2 + (blocked.y > midY);
用位元把象限編號。 (blocked.x > midX) 高位、(blocked.y > midY) 低位：
· TL = 0：blocked.x ≤ midX 且 blocked.y ≤ midY
· TR = 1：blocked.x ≤ midX 且 blocked.y > midY
· BL = 2：blocked.x > midX 且 blocked.y ≤ midY
· BR = 3：blocked.x > midX 且 blocked.y > midY 
for (int q = 0; q < 4; ++q) {
    if (q == blockedQuadrant) continue;
    middleL[count++] = center[q];
}
report(middleL);
放中央 L。 收集「不含原 blocked」3 個象限的最角落格，組成 L 直接 report()。 注意這 3 格之後在各自象限遞迴時會被當成 blocked — 它們已經被「占用」掉。 
for (int q = 0; q < 4; ++q) {
    Cell nextBlocked =
        (q == blockedQuadrant) ? blocked : center[q];
    tile(origin[q].x, origin[q].y, half, nextBlocked);
}
遞迴 4 個象限。 blocked 落在哪個象限，那象限就用原 blocked遞迴； 其他 3 個象限則用剛放下的角落格當 new special。 這保證所有 4 個子問題都是「方板 + 1 special」的同形式。
// Common Pitfalls
  • 座標 1-indexed：題目給 (X, Y) 從 1 開始，tile 的初始呼叫要傳 tile(1, 1, N, X, Y)。
  • midX = top + half − 1：是上半的最後一列，不是中線；下半從 midX + 1 開始。
  • Base case 不需要再判斷象限 — n=2 時 4 格枚舉即可。
  • 本題不需 cin/cout，ios::sync_with_stdio 純粹習慣放著（若有額外讀檔再用）。
