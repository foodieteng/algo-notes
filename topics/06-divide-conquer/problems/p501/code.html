01

完整實作// Full C++

p501.cpp C++17 
#include <bits/stdc++.h>
using namespace std;

// 回傳長度 n 的 good sequence (1..n 的合法排列)
// 規則：⌈n/2⌉ 個奇數值在前段、⌊n/2⌋ 個偶數值在後段
vector<int> build(int n) {
    if (n == 1) {
        return {1};                            // base case
    }

    int oddSize  = (n + 1) / 2;                // ⌈n/2⌉ — 奇數段長度
    int evenSize = n / 2;                      // ⌊n/2⌋ — 偶數段長度

    // 兩半各自遞迴解出 1..k 的合法排列
    vector<int> odd  = build(oddSize);
    vector<int> even = build(evenSize);

    vector<int> ans;
    ans.reserve(n);                              // 預配空間，免 push_back 重配

    // 左段映射：x → 2x − 1   (1..k 變成 1, 3, 5, ...)
    for (int x : odd) {
        ans.push_back(2 * x - 1);
    }

    // 右段映射：x → 2x        (1..k 變成 2, 4, 6, ...)
    for (int x : even) {
        ans.push_back(2 * x);
    }

    return ans;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<int> ans = build(n);

    for (int i = 0; i < n; i++) {
        cout << ans[i] << (i + 1 == n ? '\n' : ' ');
    }

    return 0;
}

實作要點：build(n) 回傳 vector<int>， 兩半各自遞迴拿到子排列 odd / even，再用 range-for 逐個映射 2x − 1、2x 推進 ans。ans.reserve(n) 預先配好容量、避免 push_back 過程多次 realloc。 整支程式沒有全域 buffer，邏輯一眼看完。

02

複雜度分析// Complexity Derivation

把 build(n) 一層的工作量拆四段：

步驟內容單層成本
遞迴左半解一個 ⌈n/2⌉ 的子問題，回傳 odd vectorT(⌈n/2⌉)
遞迴右半解一個 ⌊n/2⌋ 的子問題，回傳 even vectorT(⌊n/2⌋)
映射寫入左段range-for 把 odd 的 x 映成 2x − 1 推進 ansO(⌈n/2⌉)
映射寫入右段range-for 把 even 的 x 映成 2x 推進 ansO(⌊n/2⌋)
T(N) = T(⌈N/2⌉) + T(⌊N/2⌋) + O(N)
合併（映射＋複製）每層 O(N)，共 log N 層 ⇒ T(N) = O(N log N)
項目數量備註
TimeO(N log N)每層 O(N) 映射，共 log N 層
SpaceO(N log N)遞迴堆疊中同時存活 O(log N) 個子 vector，加總 O(N)；最終 ans 還是 O(N)
Recursion DepthO(log N)N = 10⁶ ⇒ 約 20 層，遠在 stack 安全範圍
Constant factor低vector 配置 vs. 全域陣列略慢一點，但程式碼可讀性顯著提升
03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
N = 1 直接收尾 遞迴 base 寫 if (n == 1) return {1};。若漏掉，build(1) 會繼續切成 oddSize = 1, evenSize = 0 ⇒ build(0) 又再切 oddSize = 0... 無窮遞迴。顯式 base 必加。
oddSize 用 ⌈N/2⌉ 不是 ⌊N/2⌋ 1..N 中奇數有 ⌈N/2⌉ 個（N = 5 時奇數 1, 3, 5 共 3 個）；偶數才是 ⌊N/2⌋ 個。寫反會在 N 為奇數時漏一格。
vector 傳回 vs 全域陣列 vector 版每層各自配置，遞迴展開時最壞同時存活 O(log N) 個子 vector。N = 10⁶ ⇒ 約 20 層、加總仍 O(N)，配置成本可接受。極致追求速度才需要改全域 buffer。
記得 ans.reserve(n) 沒 reserve 也對，但 push_back 過程會觸發多次容量加倍 + 搬移 ⇒ 常數變慢。預先 reserve 一行直接除掉這個開銷。
輸出格式 — 空白與換行 題目要求數字之間以空白分隔、結尾換行。cout << ans[i] << (i + 1 == n ? '\n' : ' ') 一行搞定。endl 會 flush，慢。
I/O 速度 ios::sync_with_stdio(false); cin.tie(nullptr); 必加 — 即使本題 N 不大，養成習慣。
04

細節走讀// Section Walkthrough

① base case (n = 1)

// n == 1
return {1}
直接回傳長度 1 的 vector — 沒有遞迴往下，stack 在這停。
所有遞迴展開後，葉節點都會走到這裡。

② 切半：奇數段 / 偶數段

// split
oddSize  = (n + 1) / 2     ← ⌈n/2⌉  奇數段長度
evenSize = n / 2           ← ⌊n/2⌋  偶數段長度

例：n = 5 ⇒ oddSize = 3, evenSize = 2
    n = 4 ⇒ oddSize = 2, evenSize = 2
    n = 3 ⇒ oddSize = 2, evenSize = 1

③ 兩半各自遞迴

// recurse
vector<int> odd  = build(oddSize)   ← 1..oddSize 的合法排列
vector<int> even = build(evenSize)  ← 1..evenSize 的合法排列

④ 映射：左段奇、右段偶，依序推進 ans

// map & concat
ans.reserve(n)             ← 一次配好空間

for x in odd:
    ans.push_back(2 * x - 1)   ← 1..oddSize  → 1, 3, 5, ...

for x in even:
    ans.push_back(2 * x)       ← 1..evenSize → 2, 4, 6, ...

return ans

為何 2x − 1 / 2x 雙射成立：1..k 是連續整數， 2x − 1 把它映到 {1, 3, …, 2k−1} 全部奇數， 2x 映到 {2, 4, …, 2k} 全部偶數 — 保序、唯一、值域剛好對齊。
