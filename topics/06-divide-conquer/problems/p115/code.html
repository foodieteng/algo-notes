01

完整實作// Full C++

p115.cpp C++17 
#include <bits/stdc++.h>
using namespace std;

const long long MOD = 10000019;

long long invSum = 0;            // 累積的逆序對「兩數和」(已 mod)
vector<long long> arr, buf;      // arr 就地排序、buf 為 merge 暫存

// 合併 arr[l..m] 與 arr[m+1..r]，同時累積跨界逆序對的兩數和到 invSum
void merge(int l, int m, int r) {
    // remain = arr[l..m] 之和 (mod MOD)，左半游標前進時遞減
    long long remain = 0;

    for (int i = l; i <= m; i++) {
        remain = (remain + arr[i] % MOD) % MOD;
    }

    int i = l;
    int j = m + 1;
    int k = l;

    while (i <= m && j <= r) {
        if (arr[i] <= arr[j]) {
            // 取左半：無新逆序對；remain 扣掉 arr[i]
            buf[k] = arr[i];
            remain = (remain - arr[i] % MOD + MOD) % MOD;
            i++;
            k++;
        } else {
            // arr[i] > arr[j]：左半剩 cnt 個全 > arr[j]，全部跟它配成逆序對
            long long cnt = m - i + 1;
            long long add = (remain + arr[j] % MOD * cnt % MOD) % MOD;

            invSum = (invSum + add) % MOD;

            buf[k] = arr[j];
            j++;
            k++;
        }
    }

    while (i <= m) {
        buf[k] = arr[i];
        i++;
        k++;
    }

    while (j <= r) {
        buf[k] = arr[j];
        j++;
        k++;
    }

    for (int p = l; p <= r; p++) {
        arr[p] = buf[p];
    }
}

void mergeSort(int l, int r) {
    if (l >= r) return;

    int m = l + (r - l) / 2;

    mergeSort(l, m);
    mergeSort(m + 1, r);
    merge(l, m, r);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    arr.assign(n, 0);
    buf.assign(n, 0);

    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    mergeSort(0, n - 1);

    cout << invSum << '\n';

    return 0;
}

複雜度分析// Complexity Derivation

把 mergeSort(l, r) 一層的工作量分成三段：

步驟內容單層成本
遞迴左右兩個 N/2 子問題2 · T(N/2)
算 remain掃左半 N/2 個元素一次求和O(N)
雙指針 mergei、j 各走過所有元素一遍O(N)
跨界貢獻累加每次取右半 O(1)，最多取 N 次O(N)
T(N) = 2 · T(N/2) + O(N)
Master Theorem case 2：a = 2, b = 2, f(N) = O(N) = O(Nlogb a) ⇒ T(N) = Θ(N log N)
項目數量備註
TimeO(N log N)N = 10⁶ ⇒ 約 2 × 10⁷ 次運算
SpaceO(N)全域 buf 暫存；遞迴堆疊 O(log N)
Recursion DepthO(log N)N = 10⁶ ⇒ 約 20 層
Constant factor低純整數比較、無浮點、無 sqrt

與單純的「計逆序對數量」相比，多算的只是「每次取右時加一筆 remain + arr[j] × cnt」— 仍是 O(1)，總複雜度不變。

03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
比較永遠用原始值 mod 會破壞順序 — 不同的大數可能對映到同一餘數。arr[i] <= arr[j] 必須用未 mod 的原始值。
全程 long long (arr[j] % MOD) × (cnt % MOD) ≤ 10⁷ × 10⁶ = 10¹³ — 遠超 int 上限 (2.1 × 10⁹)。remain、add、invSum、arr / buf 都用 long long。
減法要 + MOD (remain - arr[i] % MOD + MOD) % MOD — 不加 MOD 結果會變負，導致下一步累加錯誤。
中位數用安全寫法 m = l + (r − l) / 2 而不是 (l + r) / 2 — 避免 l + r 在大 N 下溢位（本題雖 n ≤ 10⁶ 不會炸，但養成習慣）。
buf 用全域 vector + assign 把 arr / buf 宣告為全域 vector<long long>，main 一次 assign(n, 0) 預配；merge 內部不要 vector<…> tmp(size) 否則每層配 / 拋一次。
copy back 別忘 merge 完後 for (p = l..r) arr[p] = buf[p] — 不寫回的話下一層 merge 還是讀到舊資料。
剩餘單邊不算貢獻 主迴圈結束時，若左半還有剩，那些元素全部小於右半被吃光的所有元素 — 不會再產生新跨界逆序對，不要更新 invSum。
I/O 速度 ios::sync_with_stdio(false); cin.tie(nullptr); 必加；10⁶ 筆輸入用 cin/cout 沒這兩行會 TLE。輸出用 '\n'，別用 endl。
04

細節走讀// Section Walkthrough

① 切半遞迴

// split + recurse
if (l >= r) return;             ← base：單一元素，無逆序對
m = l + (r - l) / 2;
mergeSort(l,     m);            ← 左半排好、左半內部逆序對計完
mergeSort(m + 1, r);            ← 右半排好、右半內部逆序對計完
merge(l, m, r);                 ← 兩半合併 + 算「跨界」逆序對

② 初始化 remain (左半剩餘和)

// remain init
remain = 0
for i in [l, m]:
    remain = (remain + arr[i] % MOD) % MOD

意義：左半當前剩餘元素之和 (mod MOD)。
之後每從左半取出一個 arr[i]，就 remain -= arr[i]。
這樣 add 計算可以 O(1) 取得「左半剩餘和」。

③ 雙指針 merge — 取左 vs 取右

// two-pointer merge
i = l, j = m+1, k = l
while i <= m AND j <= r:
    if arr[i] <= arr[j]:             ← 比較用原始值！
        buf[k] = arr[i]
        remain -= arr[i] (mod)       ← 別忘了扣
        i++; k++
    else:                            ← arr[i] > arr[j]，是逆序對群
        cnt = m - i + 1              ← 左半剩餘個數
        add = remain + arr[j] × cnt  ← 一次累積
        invSum += add (mod)
        buf[k] = arr[j]
        j++; k++

關鍵公式：add = remain + arr[j] × cnt — 左半剩 cnt 個元素，每個跟 arr[j] 配成逆序對，那些元素的和就是 remain， arr[j] 出現 cnt 次貢獻 arr[j] × cnt。

④ 收尾 + 寫回原陣列

// drain + copy back
while i <= m:   buf[k++] = arr[i++]   ← 左半剩餘倒進 buf
while j <= r:   buf[k++] = arr[j++]   ← 右半剩餘倒進 buf

for p in [l, r]:
    arr[p] = buf[p]                    ← 寫回，上層 merge 才能讀到

為什麼收尾不算貢獻？若主迴圈結束時左半還有剩，代表右半被吃完了。那些右半元素都已比左半剩的還小（不然不會被先取出）， 所以左半剩的元素跟「已吃光的右半」沒有逆序對關係 — 在前面 if 分支已經處理過了。
