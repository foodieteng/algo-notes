完整實作// Full C++

p501.cpp C++17 
// P501 · 好的序列 — 奇偶切構造
// f(N) = [ 2·f(⌈N/2⌉) − 1 ]  ∥  [ 2·f(⌊N/2⌋) ]
#include <iostream>
using namespace std;

const int MAXN = 1000005;
int seq[MAXN];   // 主答案緩衝區，最後印出 seq[0..N-1]
int buf[MAXN];   // 暫存遞迴子結果，避免額外配置

// 把長度 len 的「1..len 的合法排列」build 到 out[0..len-1]
void build(int *out, int len) {
    if (len == 1) {                  // base case
        out[0] = 1;
        return;
    }

    int leftLen  = (len + 1) / 2;      // ⌈len/2⌉ — 奇數段長度
    int rightLen = len - leftLen;      // ⌊len/2⌋ — 偶數段長度

    // 先用 buf 解出長 leftLen 的子問題，再就地映成 2x − 1（奇）
    build(buf, leftLen);
    for (int i = 0; i < leftLen; ++i)
        out[i] = 2 * buf[i] - 1;

    if (rightLen == 0) return;           // len == 1 已 return；此處保險

    // 再用 buf 解長 rightLen 的子問題，映成 2x（偶），接在 out[leftLen..]
    build(buf, rightLen);
    for (int i = 0; i < rightLen; ++i)
        out[leftLen + i] = 2 * buf[i];
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);

    int n; cin >> n;
    build(seq, n);

    for (int i = 0; i < n; ++i)
        cout << seq[i] << (i + 1 == n ? '\n' : ' ');
    return 0;
}

實作要點：用兩個全域 int 陣列 seq[] 與 buf[]， build() 把長 len 的子問題直接寫到傳進來的 out 指標。 buf 在遞迴前後即用即覆蓋 — 因為每層先「解左」再「映射複製到 out 左半」， 接著「解右」再「映射複製到 out 右半」，左右兩次解的中間 buf 內容不需保留。

02

複雜度分析// Complexity Derivation

把 build(out, len) 一層的工作量拆三段：

步驟內容單層成本
遞迴左半解一個 ⌈len/2⌉ 的子問題T(⌈len/2⌉)
映射並寫入左段for-loop 把 buf[i] 映成 2x − 1 寫到 out[0..]O(⌈len/2⌉)
遞迴右半解一個 ⌊len/2⌋ 的子問題T(⌊len/2⌋)
映射並寫入右段for-loop 把 buf[i] 映成 2x 寫到 out[leftLen..]O(⌊len/2⌋)
T(N) = T(⌈N/2⌉) + T(⌊N/2⌋) + O(N)
合併（映射＋複製）每層 O(N)，共 log N 層 ⇒ T(N) = O(N log N)
項目數量備註
TimeO(N log N)每層 O(N) 映射，共 log N 層
SpaceO(N)seq[] + buf[] 兩支陣列；無動態配置
Recursion DepthO(log N)N = 10⁶ ⇒ 約 20 層，遠在 stack 安全範圍
Constant factor極低純加減乘、無比較、無排序、無 STL container
03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
N = 1 直接收尾 遞迴 base 寫 if (len == 1) { out[0] = 1; return; }。若忘了寫，leftLen = 1, rightLen = 0 雖然不爆，但右段 for-loop 跑 0 次 ⇒ 仍能正確 return — 不過建議保留顯式 base，可讀性高。
leftLen 用 ⌈N/2⌉ 不是 ⌊N/2⌋ 奇數段 1..N 有 ⌈N/2⌉ 個（N=5 時奇數 1,3,5 有 3 個）；偶數段才是 ⌊N/2⌋ 個。寫反會在 N 為奇數時越界或漏元素。
映射順序：先映左、再解右 buf 是共用緩衝。先 build(buf, leftLen) 然後立刻把 buf 內容映射複製到 out 左段；再 build(buf, rightLen) 覆蓋 buf，這時左段已寫入 out 不會被破壞。
輸出格式 — 空白與換行 題目要求數字之間以空白分隔、結尾換行。用 cout << seq[i] << (i + 1 == n ? '\n' : ' ') 一行搞定。endl 會 flush，慢。
0-indexed vs 1-indexed 數值 1..N 與下標 0..N−1 是兩件事 — code 用 0-indexed 跑陣列，輸出本身就是值 1..N，不需 +1。
I/O 速度 ios::sync_with_stdio(false); cin.tie(nullptr); 必加 — 即使本題 N 不大，養成習慣。
04

細節走讀// Section Walkthrough

① base case (len = 1)

// len == 1
out[0] = 1
直接 return — 沒有遞迴往下，stack 在這停。
所有遞迴展開後，葉節點都會走到這裡。

② 切半：奇數段 / 偶數段

// split
leftLen  = (len + 1) / 2     ← ⌈len/2⌉  奇數段
rightLen = len - leftLen     ← ⌊len/2⌋  偶數段

例：len = 5 ⇒ leftLen = 3, rightLen = 2
    len = 4 ⇒ leftLen = 2, rightLen = 2
    len = 3 ⇒ leftLen = 2, rightLen = 1

③ 遞迴解左半並映射成奇數

// build left half (奇數段)
build(buf, leftLen)            ← 解出 1..leftLen 的合法排列到 buf[0..leftLen-1]
for i in [0, leftLen):
    out[i] = 2 * buf[i] - 1    ← 把 1..leftLen 映成 1, 3, 5, ...

映射 2x − 1 把整數 1..⌈N/2⌉ 雙射到奇數 1, 3, …, 2·⌈N/2⌉−1，保序且唯一。

④ 遞迴解右半並映射成偶數，接在後段

// build right half (偶數段)
build(buf, rightLen)           ← 重新解出 1..rightLen 的排列，覆蓋 buf
for i in [0, rightLen):
    out[leftLen + i] = 2 * buf[i]   ← 映成 2, 4, 6, ... 接到 out 後段

buf 在「解左→映左」與「解右→映右」之間可以被覆蓋，因為左段已寫入 out。 這就是用單支共用緩衝即可、不需要遞迴內配置記憶體的關鍵。
