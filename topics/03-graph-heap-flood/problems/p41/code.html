庭院裡的水池

01

完整實作// Full C++

p41.cpp C++17 
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1005;

char grid[MAXN][MAXN];          // 地圖：'.' 水 / '#' 障礙
bool visited[MAXN][MAXN];       // 這格是否已被某灘淹過

// BFS 用的手動 queue（int 陣列，不吃呼叫堆疊）
int qx[MAXN * MAXN];
int qy[MAXN * MAXN];

// 上下左右四方向
int dx[4] = {-1, 1, 0, 0};
int dy[4] = {0, 0, -1, 1};

int H, W;

// 從 (sx, sy) 出發，BFS 把整灘水標記為 visited
void flood(int sx, int sy) {
    int head = 0, tail = 0;
    qx[tail] = sx;
    qy[tail] = sy;
    tail++;
    visited[sx][sy] = true;           // 進 queue 當下就標記，避免重複入列

    while (head < tail) {
        int x = qx[head];
        int y = qy[head];
        head++;

        for (int d = 0; d < 4; d++) {
            int nx = x + dx[d];
            int ny = y + dy[d];

            if (nx < 0 || nx >= H || ny < 0 || ny >= W) continue;  // 出界
            if (grid[nx][ny] != '.') continue;            // 障礙
            if (visited[nx][ny]) continue;                // 走過

            visited[nx][ny] = true;
            qx[tail] = nx;
            qy[tail] = ny;
            tail++;
        }
    }
}

void solve() {
    cin >> H >> W;

    for (int i = 0; i < H; i++) {
        cin >> grid[i];                        // 一整行字串讀入
        for (int j = 0; j < W; j++) visited[i][j] = false;
    }

    int pools = 0;

    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++) {
            if (grid[i][j] == '.' && !visited[i][j]) {
                pools++;                          // 沒走過的水 ⇒ 新的一灘
                flood(i, j);                     // 一次淹滿、全部標記
            }
        }
    }

    cout << pools << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;
    while (T--) solve();

    return 0;
}

為何用 BFS 而非遞迴 DFS？ 1000×1000 全是水時，遞迴 DFS 深度可達 10⁶ — 預設呼叫堆疊撐不住。 手動 queue 把待辦格子放在全域陣列（heap 記憶體），深度再大也不爆。

02

複雜度分析// Complexity Derivation

單筆測資拆成兩段：主迴圈掃描 + 所有 BFS 加總。

步驟內容成本
主迴圈掃描走過 H×W 格各一次，判斷是否開新池O(H × W)
BFS 入列每格水最多入列一次（入列即標 visited）O(H × W)
BFS 出列 + 看四方向每格出列一次，常數 4 個鄰居O(4 · H × W)
T = O(H × W) + O(H × W) = O(H × W)
每格被造訪 O(1) 次 ⇒ 全圖線性；多筆乘上 T ≤ 10
項目數量備註
TimeO(H × W)1000×1000×10 ≈ 10⁷，秒過
SpaceO(H × W)grid + visited + queue 各約 10⁶
遞迴深度0（迭代）BFS 用全域陣列，不吃呼叫堆疊
Constant factor低整數索引、4 方向 if
03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
遞迴 DFS 爆 stack 全水 1000×1000 ⇒ 遞迴深 10⁶。改用 BFS / 迭代 DFS，待辦放全域陣列。
入列時才標 visited 若等「出列時」才標，同一格可能被多個鄰居重複推進 queue ⇒ queue 爆、重複工作。入列當下就標。
多筆測資要清 visited 每筆重讀地圖時把 visited 歸零（這裡逐格清，只清用到的範圍）。
queue 開夠大 最壞整圖全水 ⇒ qx/qy 要能裝 H × W 個元素，開 MAXN*MAXN。
讀整行字串 cin >> grid[i] 把一整列讀進 char 陣列；別用逐字 getchar 混到換行。
只連 4 方向 題目是上下左右相鄰才同灘，不含對角 ⇒ dx/dy 只放 4 個方向。
04

細節走讀// Section Walkthrough

① 主迴圈 — 找新池的觸發點

// main scan
for i in [0, H):
  for j in [0, W):
    if grid[i][j] == '.' and not visited[i][j]:
        pools++          ← 只有「沒走過的水」會 +1
        flood(i, j)      ← 立刻淹滿整灘，封住這片

② flood — BFS 用手動 queue

// flood(sx, sy)
push (sx, sy); visited[sx][sy] = true   ← 入列即標記
while queue 非空:
    (x, y) = pop front
    for d in 4 方向:
        (nx, ny) = (x+dx[d], y+dy[d])
        略過: 出界 / 障礙 / 已走過
        否則 visited[nx][ny]=true; push (nx,ny)

head 是隊頭、tail 是隊尾；head < tail 代表 queue 還有東西。 指標只往前走、不回頭 ⇒ 每格 push 一次、pop 一次。

③ 為何「入列即標記」是關鍵

// 反例：若改成出列才標記
某格 (a,b) 同時是兩個已入列鄰居的鄰居
→ 兩次都判定「未 visited」→ 被 push 兩次
→ queue 膨脹、同格處理兩次（雖不影響答案，但可能爆陣列）
∴ push 進去的當下就 visited[ ] = true
