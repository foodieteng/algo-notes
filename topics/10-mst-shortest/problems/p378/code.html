01

完整實作// Full C++

p378.cpp C++17 
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll INF = (ll)4e18;

// 在圖 g 上從 src 跑 Dijkstra，結果寫進 dist
void dijkstra(int src, vector<vector<pair<int, ll>>>& g, vector<ll>& dist) {
    dist.assign(g.size(), INF);
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<>> pq;
    dist[src] = 0;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;
        for (auto [v, w] : g[u])
            if (d + w < dist[v]) { dist[v] = d + w; pq.push({dist[v], v}); }
    }
}

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M, Q;
    cin >> N >> M >> Q;

    vector<vector<pair<int, ll>>> g(N + 1), gr(N + 1);   // 正圖、反圖
    for (int i = 0; i < M; i++) {
        int a, b; ll w;
        cin >> a >> b >> w;
        g[a].push_back({b, w});          // a → b
        gr[b].push_back({a, w});         // 反圖：b → a
    }

    vector<ll> d1, dN;
    dijkstra(1, g,  d1);     // d1[v] = 1 到 v 的最短距離
    dijkstra(N, gr, dN);     // dN[v] = v 到 N 的最短距離（反圖上 N 到 v）

    ll base = d1[N];            // 不加任何邊的原最短路

    while (Q--) {
        int c, d;
        cin >> c >> d;

        ll cand = INF;
        if (d1[c] < INF && dN[d] < INF)
            cand = d1[c] + 1 + dN[d];   // 1 ⇝ c → d ⇝ N

        cout << min(base, cand) << '\n';
    }
}

int main() { solve(); return 0; }

建反圖只要在讀每條邊時順手把 (a,b,w) 反著塞進 gr。兩次 Dijkstra 共用同一個函式。

複雜度分析// Complexity Derivation

拆成預處理 + 查詢兩段：

步驟內容成本
建圖讀 M 條邊，同時建正圖 + 反圖O(M)
正向 Dijkstra從 1 求 d1[]O(M log N)
反向 Dijkstra從 N 在反圖求 dN[]O(M log N)
每個 query一次加法 + minO(1)
T = O(M log N + Q)
兩次 Dijkstra 預處理，Q 個查詢各 O(1)
項目數量備註
TimeO(M log N + Q)M=5×10⁵ ⇒ ~10⁷
SpaceO(N + M)正圖 + 反圖 + d1[] + dN[]
數值型別long long路徑和 ~2×10¹⁴；INF 設 4×10¹⁸
查詢O(1)預處理後純查表
03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
query 各自獨立 候選邊只是「假設加上去」，不會累積到下一個 query。千萬別把 Q 條邊一起塞進圖再跑一次。
反圖求 dN[] 軌道是有向的。「v 到 N 的距離」要在反圖上從 N 出發跑，不能在正圖從 N 跑（方向相反）。
候選邊長度固定 1 不論 c、d 是誰，新邊長度都是 1，公式中間那項是常數 1。
不可達要擋掉 若 d1[c] 或 dN[d] 為 INF，d1[c]+1+dN[d] 會溢位或變垃圾值。先判 < INF 再相加。
long long + INF 夠大 w ≤ 10⁹、路徑 ~2×10⁵ 段 ⇒ 距離 ~2×10¹⁴。INF 取 4×10¹⁸，避免「INF+1」溢位 long long。
I/O 速度 M、Q 都很大 ⇒ sync_with_stdio(false); cin.tie(nullptr); 必加，輸出用 '\n'。
04

細節走讀// Section Walkthrough

① 建正圖 + 反圖

// build both graphs
for each edge (a, b, w):
    g[a].push_back({b, w})     ← 正圖 a→b
    gr[b].push_back({a, w})    ← 反圖 b→a（同一條邊反著存）

② 兩次 Dijkstra 預處理

// preprocess
dijkstra(1, g,  d1)   ← d1[v] = 起點 1 到 v
dijkstra(N, gr, dN)   ← dN[v] = v 到終點 N（反圖上 N 到 v）
base = d1[N]          ← 原本的最短路

③ 每個 query O(1)

// per query (c, d)
cand = INF
if d1[c] < INF and dN[d] < INF:
    cand = d1[c] + 1 + dN[d]   ← 1 ⇝ c → d ⇝ N
ans = min(base, cand)          ← 用 or 不用候選邊
print ans

④ 為何 dN 要在反圖跑

// why reverse graph
原圖「d 到 N」的最短路
 = 反圖「N 到 d」的最短路   ← 把每條邊反向，起終點互換
所以從 N 在反圖跑一次，就一次拿到所有 dN[d]。
