01

完整實作// Full C++

p151.cpp C++17 
#include <bits/stdc++.h>
using namespace std;

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;
    while (T--) {
        int n, m;
        cin >> n >> m;

        // adj[a] 收所有「攻 a 之後才解鎖」的陣地；indeg[b] = b 還欠幾個前置
        vector<vector<int>> adj(n);
        vector<int> indeg(n, 0);

        for (int i = 0; i < m; i++) {
            int a, b;
            cin >> a >> b;                       // 攻 b 前須先攻 a：邊 a → b
            adj[a].push_back(b);
            indeg[b]++;
        }

        // min-heap：每次取「目前可選中編號最小」的陣地 ⇒ 字典序最小
        priority_queue<int, vector<int>, greater<int>> pq;
        for (int v = 0; v < n; v++)
            if (indeg[v] == 0) pq.push(v);

        vector<int> order;
        order.reserve(n);
        while (!pq.empty()) {
            int u = pq.top();
            pq.pop();
            order.push_back(u);

            for (int w : adj[u]) {
                if (--indeg[w] == 0) pq.push(w);  // 解鎖 ⇒ 進堆
            }
        }

        // 輸出不滿 n 個 ⇒ 剩下的點卡在循環裡 ⇒ 無解
        if ((int)order.size() != n) {
            cout << "QAQ\n";
        } else {
            for (int i = 0; i < n; i++)
                cout << order[i] << " \n"[i == n - 1];  // 最後一個接換行
        }
    }
}

int main() { solve(); return 0; }

" \n"[i == n-1] 是個小技巧：非最後元素印空格（索引 0），最後元素印換行（索引 1）。

複雜度分析// Complexity Derivation

把單組測資的工作量拆開：

步驟內容成本
建圖讀 m 條邊、填 adj 與 indegO(n + m)
初始入堆掃 n 個點，indeg=0 者 pushO(n log n)
主迴圈取點每點 pop 一次O(n log n)
鬆弛入度每條邊掃一次，可能觸發 pushO(m log n)
T = O((n + m) log n)
每個點進出堆各一次、每條邊掃一次；log 來自 priority_queue
項目數量備註
TimeO((n+m) log n)n=10⁵, m=3×10⁵ ⇒ ~7×10⁶ 操作
SpaceO(n + m)adj 表 O(n+m)、indeg O(n)、堆 O(n)
每組測資重建× T (≤ 5)vector 在迴圈內宣告，自動清空
Constant factor中堆操作常數略高但量級內輕鬆
03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
邊的方向 「攻 b 前先攻 a」⇒ 邊是 a → b，indeg[b]++。方向反了會得到反序、整個錯。
必須用 min-heap 一般 queue 只給「某個」合法序；要字典序最小一定要 priority_queue<int, vector<int>, greater<int>>（小根堆）。
判循環的方式 不必額外偵測環。輸出個數 ≠ n 就代表有點永遠進不了堆（卡在環中）⇒ 輸出 QAQ。
多筆測資清空 T 組各自獨立。把 adj、indeg 宣告在 while(T--) 內，每組自動是乾淨的。
m 可能為 0 沒有任何依賴時，所有點 indeg=0、全進堆 ⇒ 輸出就是 0 1 2 … n−1，程式自然處理。
I/O 速度 m 達 3×10⁵、再 ×5 組 ⇒ 必加 sync_with_stdio(false); cin.tie(nullptr);，輸出用 '\n' 別用 endl。

細節走讀// Section Walkthrough

① 建圖 + 算入度

// build graph
for each edge (a, b):     ← 攻 b 前須先攻 a
    adj[a].push_back(b)   ← a 完成後，b 少欠一個前置
    indeg[b]++            ← b 的「欠債數」+1

② 初始化 min-heap

// seed heap
for v in [0, n):
    if indeg[v] == 0:     ← 沒有前置，現在就能打
        pq.push(v)
pq 是小根堆 ⇒ top() 永遠是目前可選的「最小編號」

③ 反覆取最小 + 鬆弛

// main loop
while pq not empty:
    u = pq.top(); pq.pop()       ← 取目前可選的最小
    order.push_back(u)           ← 加入答案
    for w in adj[u]:
        if --indeg[w] == 0:      ← w 的前置剛好全打完
            pq.push(w)           ← 解鎖 ⇒ 進堆候選

④ 判合法 / 輸出

// finalize
if order.size() != n:    ← 有點進不了堆 ⇒ 卡在循環
    print "QAQ"
else:
    print order[0..n-1]  ← 用空格隔開、結尾換行
