最小生成樹

完整實作 · Kruskal + Prim// Two Implementations

p472.cpp C++17 
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

// ── 並查集（路徑壓縮 + 按秩合併）
int par[200005], rnk[200005];

int find(int x) {
    while (par[x] != x) {
        par[x] = par[par[x]];      // 路徑壓縮（半路徑）
        x = par[x];
    }
    return x;
}

bool unite(int a, int b) {
    a = find(a); b = find(b);
    if (a == b) return false;             // 已同塊 ⇒ 會成環
    if (rnk[a] < rnk[b]) swap(a, b);
    par[b] = a;
    if (rnk[a] == rnk[b]) rnk[a]++;
    return true;
}

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    // 邊存成 {w, u, v}，按權重由小到大排序（array 的字典序就先比 w）
    vector<array<ll, 3>> edges(M);
    for (int i = 0; i < M; i++) {
        ll u, v, w;
        cin >> u >> v >> w;
        edges[i] = {w, u, v};
    }
    sort(edges.begin(), edges.end());

    for (int i = 1; i <= N; i++) { par[i] = i; rnk[i] = 0; }

    // 由小到大貪心加邊，只取「連接兩個不同連通塊」的邊
    ll total = 0;
    int used = 0;
    for (auto& e : edges) {
        ll w = e[0]; int u = e[1], v = e[2];
        if (unite(u, v)) {
            total += w;
            if (++used == N - 1) break;      // 已連成生成樹，提早結束
        }
    }

    cout << total << '\n';
}

int main() { solve(); return 0; }

把邊存成 {w, u, v} 順序，array 的字典序比較剛好先比 w ⇒ 一個 sort 就排好權重。

另解 Prim（min-heap）—— 從一點長出樹

Kruskal 以邊為中心（排序 + 並查集判環）；Prim 以點為中心： 從任一點出發，反覆用 min-heap 取「離當前樹最近的點」把它併進來，直到 N 個點全進樹。

p472_prim.cpp C++17 
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    // 鄰接表：無向圖，每條邊雙向加入
    vector<vector<pair<int, ll>>> adj(N + 1);
    for (int i = 0; i < M; i++) {
        int u, v; ll w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});          // 無向 ⇒ 兩邊都加
    }

    // Prim：從點 1 長出樹，min-heap 取「離樹最近的點」
    vector<char> inTree(N + 1, 0);
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<>> pq;
    pq.push({0, 1});                       // (連到樹的成本, 點)

    ll total = 0;
    int cnt = 0;
    while (!pq.empty() && cnt < N) {
        auto [w, u] = pq.top(); pq.pop();
        if (inTree[u]) continue;            // 過期條目：u 已在樹裡，跳過
        inTree[u] = 1;
        total += w;                        // 把連 u 進來的那條邊計入
        cnt++;
        for (auto [v, ew] : adj[u])
            if (!inTree[v]) pq.push({ew, v});  // 鬆弛：u 的鄰居入堆
    }

    cout << total << '\n';
}

int main() { solve(); return 0; }

lazy 版 Prim：不更新堆裡的舊值，靠 if (inTree[u]) continue; 略過過期條目。 圖保證連通 ⇒ 一定長到 N 個點。本題給的是邊表，Kruskal 寫起來更短；Prim 在稠密圖或鄰接表已建好時更順手。

面向KruskalPrim（min-heap）
中心思想以邊為主：排序 + 並查集判環以點為主：從一點長出樹
主要結構邊陣列 + DSU鄰接表 + min-heap
複雜度O(M log M)O(M log N)
適合稀疏圖、給定邊表稠密圖、已有鄰接表
判環/重複find(u)==find(v) 跳過inTree[u] 跳過過期條目

03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
權和開 long long 最多 N−1 ≈ 2×10⁵ 條邊、每條 w ≤ 10⁹ ⇒ 權和達 2×10¹⁴，遠超 int。total 必用 ll。
DSU 不優化會 TLE/退化 沒有路徑壓縮 + 按秩，find 最壞 O(N)、整體退化。兩個優化都要加。
判環靠 find 比較 加邊前先 find(u)==find(v) ⇒ 同塊就跳過（成環）。這就是 Kruskal 的核心過濾。
提早結束 加滿 N−1 條邊就能 break，省下後面的邊。圖保證連通 ⇒ 一定湊得滿。
無向邊 MST 是無向圖概念，unite(u,v) 不分方向，一條邊處理一次即可。
I/O 速度 M 達 2×10⁵ ⇒ sync_with_stdio(false); cin.tie(nullptr); 必加，輸出用 '\n'。

③ 貪心加邊（核心）

// kruskal main
total = 0; used = 0
for (w, u, v) in edges:        ← 已由小到大
    if find(u) != find(v):     ← 不同塊 ⇒ 加它不成環
        unite(u, v)
        total += w
        if ++used == N-1: break   ← 連滿就停
    else:
        skip                   ← 同塊，加了會成環

④ find / unite 細節

// path compression + union by rank
find: while par[x]!=x: par[x]=par[par[x]]; x=par[x]   ← 半路徑壓縮
unite: 把矮的接到高的；同高則被接者 rank+1
