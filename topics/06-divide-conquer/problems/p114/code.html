Clean Code

p114.cpp C++ 
#include <iostream>
#include "lib0114.h"
using namespace std;

// 在以 (r, c) 為左上角、邊長為 n 的子方格內鋪 L-tromino，
// special cell 位於 (sx, sy)。
void tile(int r, int c, int n, int sx, int sy) {
    if (n == 2) {
        // 4 格中除 special 外的 3 格本身就是一個 L
        int xs[3], ys[3], k = 0;
        for (int dr = 0; dr < 2; ++dr) {
            for (int dc = 0; dc < 2; ++dc) {
                int x = r + dr, y = c + dc;
                if (x == sx && y == sy) continue;
                xs[k] = x; ys[k] = y; ++k;
            }
        }
        Report(xs[0], ys[0], xs[1], ys[1], xs[2], ys[2]);
        return;
    }

    int h  = n / 2;
    int mr = r + h - 1;     // 上半最後一列
    int mc = c + h - 1;     // 左半最後一行

    // 4 個象限「最靠近中心」的格子座標
    int qx[4] = { mr, mr,     mr + 1, mr + 1 };
    int qy[4] = { mc, mc + 1, mc,     mc + 1 };

    // special 落在哪個象限？  0=TL, 1=TR, 2=BL, 3=BR
    int sq = (sx > mr) * 2 + (sy > mc);

    // 在中央放 L：3 cell = 不含 special 的 3 個象限的最角落
    int rx[3], ry[3], k = 0;
    for (int q = 0; q < 4; ++q) {
        if (q == sq) continue;
        rx[k] = qx[q]; ry[k] = qy[q]; ++k;
    }
    Report(rx[0], ry[0], rx[1], ry[1], rx[2], ry[2]);

    // 對 4 象限分別遞迴；非 special 象限以「剛放下的角落格」當新 special
    int sr[4] = { r,     r,     r + h, r + h };
    int sc[4] = { c,     c + h, c,     c + h };
    for (int q = 0; q < 4; ++q) {
        int nx = (q == sq) ? sx : qx[q];
        int ny = (q == sq) ? sy : qy[q];
        tile(sr[q], sc[q], h, nx, ny);
    }
}

void solve(int N, int X, int Y) {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    tile(1, 1, N, X, Y);
}

逐段走讀

核心是 tile(r, c, n, sx, sy)：對一個左上角 (r,c)、邊長 n、缺一格 (sx,sy) 的子板鋪 L。

if (n == 2) {
    /* 4 格扣掉 special 後 Report */
}
Base case。 2×2 板只有 4 格，扣掉 special 後剩 3 格必然構成 L。 直接掃 4 格，跳過 special，剩 3 格送進 Report。 
int mr = r + h - 1;
int mc = c + h - 1;
int qx[4] = { mr, mr, mr+1, mr+1 };
int qy[4] = { mc, mc+1, mc, mc+1 };
中央 4 角座標。 每個象限「最靠近中心」的格子，按 TL / TR / BL / BR 順序排： (mr,mc)、(mr,mc+1)、(mr+1,mc)、(mr+1,mc+1)。 這 4 格就是放置中央 L 的候選。 
int sq = (sx > mr) * 2 + (sy > mc);
用位元把象限編號。 (sx>mr) 高位、(sy>mc) 低位：
· TL = 0：sx ≤ mr 且 sy ≤ mc
· TR = 1：sx ≤ mr 且 sy > mc
· BL = 2：sx > mr 且 sy ≤ mc
· BR = 3：sx > mr 且 sy > mc 
for (int q = 0; q < 4; ++q) {
    if (q == sq) continue;
    rx[k] = qx[q]; ry[k] = qy[q]; ++k;
}
Report(rx[0], ry[0], rx[1], ry[1],
       rx[2], ry[2]);
放中央 L。 收集「不含 special」3 個象限的最角落格，組成 L 直接 Report。 注意這 3 格之後在各自象限遞迴時會被當成 special — 它們已經被「占用」掉。 
for (int q = 0; q < 4; ++q) {
    int nx = (q == sq) ? sx : qx[q];
    int ny = (q == sq) ? sy : qy[q];
    tile(sr[q], sc[q], h, nx, ny);
}
遞迴 4 個象限。 special 落在哪個象限，那象限就用原 special遞迴； 其他 3 個象限則用剛放下的角落格當 new special。 這保證所有 4 個子問題都是「方板 + 1 special」的同形式。
// Common Pitfalls
  • 座標 1-indexed：題目給 (X, Y) 從 1 開始，tile 的初始呼叫要傳 tile(1, 1, N, X, Y)。
  • mr = r + h − 1：是上半的最後一列，不是中線；下半從 mr + 1 開始。
  • Base case 不需要再判斷 sq — n=2 時 4 格枚舉即可。
  • 本題不需 cin/cout，ios::sync_with_stdio 純粹習慣放著（若有額外讀檔再用）。
