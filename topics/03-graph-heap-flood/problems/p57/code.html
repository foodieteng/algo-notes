完整實作 · 手刻 min-heap// Full C++

p57.cpp C++17 
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1000006;

int heap[MAXN];      // 1-indexed 完全二元樹；heap[1] 永遠是最小值
int sz = 0;          // 目前元素個數

// 插入：放尾端，往上浮到不比父小為止
void push(int v) {
    heap[++sz] = v;
    int i = sz;

    while (i > 1 && heap[i] < heap[i / 2]) {   // 比父小 ⇒ 上浮
        swap(heap[i], heap[i / 2]);
        i /= 2;
    }
}

// 取出最小：根與尾端互換、size--，再把新的根往下沉
int pop() {
    int ret = heap[1];            // 最小值
    heap[1] = heap[sz--];        // 最後一個搬到根，元素數 -1

    int i = 1;
    while (true) {
        int smallest = i;
        int l = i * 2, r = i * 2 + 1;

        if (l <= sz && heap[l] < heap[smallest]) smallest = l;
        if (r <= sz && heap[r] < heap[smallest]) smallest = r;

        if (smallest == i) break;       // 父 ≤ 兩子 ⇒ 定位完成

        swap(heap[i], heap[smallest]);  // 與較小子交換，往下沉
        i = smallest;
    }

    return ret;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;

    while (T--) {
        int op;
        cin >> op;

        if (op == 1) {
            int n;
            cin >> n;
            push(n);
        } else {                       // op == 2 ⇒ pop-min
            if (sz == 0) cout << "empty!\n";
            else          cout << pop() << '\n';
        }
    }

    return 0;
}

偷懶版：STL 直接給你 heap — priority_queue<int, vector<int>, greater<int>> pq; 就是 min-heap，pq.push(n) / pq.top() / pq.pop()。本卡示範手刻是為了理解內部； 競賽時用 priority_queue 即可。

複雜度分析// Complexity Derivation

步驟內容成本
push · sift-up從尾端往父走，最多走樹高層O(log n)
pop · sift-down從根往較小子走，最多走樹高層O(log n)
每層比較 / 交換常數次（看 1 個父或 2 個子）O(1)
總時間 = T × O(log n) = O(T log T)
T = 10⁶ ⇒ ~2×10⁷ 次運算；I/O 量大，務必關 sync
項目數量備註
TimeO(T log T)每操作 O(log n)
SpaceO(n)heap[] 最壞裝下全部 push 進來的數
遞迴深度0（迭代）sift 用 while 迴圈，無遞迴
Constant factor低整數比較、index 乘除 2
03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
用 1-indexed 父子公式 2i / 2i+1 / i/2 才漂亮。若 0-indexed 要改成 2i+1 / 2i+2 / (i−1)/2，易錯。
sift-down 要比「較小的子」 min-heap 下沉時跟兩個子裡較小的那個換，不是隨便挑左子；否則破壞 heap 性質。
子節點存在性檢查 取 l, r 前要 l <= sz / r <= sz，否則讀到界外或已 pop 的舊值。
空 heap pop sz == 0 時輸出 empty!，不要進 pop（會讀 heap[1] 垃圾值 / sz 變負）。
10⁶ I/O 要加速 ios::sync_with_stdio(false); cin.tie(nullptr); 必加；輸出用 '\n' 不用 endl。
陣列開夠大 最壞全是 push ⇒ heap[] 要能放 T = 10⁶ 個 ⇒ MAXN ≈ 10⁶ + 餘裕。
04

細節走讀// Section Walkthrough

① push — sift-up（上浮）

// push(v)
heap[++sz] = v            ← 放到陣列尾端（樹的最後一個位置）
i = sz
while i > 1 and heap[i] < heap[i/2]:   ← 比父小
    swap(heap[i], heap[i/2])           ← 上浮
    i /= 2
// 直到 ≥ 父 或 到根 ⇒ heap 性質恢復

② pop — sift-down（下沉）

// pop()
ret = heap[1]             ← 根 = 最小值
heap[1] = heap[sz--]      ← 最後一個搬到根，size 減 1
i = 1
loop:
    smallest = i
    l = 2i, r = 2i+1
    if l<=sz and heap[l]<heap[smallest]: smallest = l
    if r<=sz and heap[r]<heap[smallest]: smallest = r
    if smallest == i: break    ← 父已 ≤ 兩子
    swap(heap[i], heap[smallest]); i = smallest   ← 與較小子換、下沉
return ret

為何取「最後一個搬到根」？ 這樣才維持完全二元樹的形狀（最後一層由右往左收）； 再用 sift-down 把這個可能很大的值沉到正確位置。

③ STL 對照

// priority_queue 版（min-heap）
priority_queue<int, vector<int>, greater<int>> pq;
op 1 n → pq.push(n);
op 2   → if (pq.empty()) cout << "empty!\n";
         else { cout << pq.top() << '\n'; pq.pop(); }
