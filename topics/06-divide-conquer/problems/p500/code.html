完整實作// Full C++

p500.cpp C++17 
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 500005;

int n;
int a[MAXN];                 // 1-indexed 排列
int lMax[MAXN], lMin[MAXN];  // a[l..mid] 的 max/min，以 l 為索引
int rMax[MAXN], rMin[MAXN];  // a[mid+1..r] 的 max/min，以 r 為索引
int bucket[2 * MAXN];        // case 3、4 共用的計數桶（用完清乾淨，跨測資沿用）

void buildPrefix(int lo, int mid, int hi) {
    lMax[mid] = lMin[mid] = a[mid];
    for (int l = mid - 1; l >= lo; --l) {
        lMax[l] = max(lMax[l + 1], a[l]);
        lMin[l] = min(lMin[l + 1], a[l]);
    }
    rMax[mid + 1] = rMin[mid + 1] = a[mid + 1];
    for (int r = mid + 2; r <= hi; ++r) {
        rMax[r] = max(rMax[r - 1], a[r]);
        rMin[r] = min(rMin[r - 1], a[r]);
    }
}

// Case 1：max、min 都在左 → good ⟺ r = l + (lMax[l] − lMin[l])
long long maxLminL(int lo, int mid, int hi) {
    long long good = 0;
    for (int l = lo; l <= mid; ++l) {
        int r = l + lMax[l] - lMin[l];
        if (r > mid && r <= hi && rMax[r] < lMax[l] && rMin[r] > lMin[l]) ++good;
    }
    return good;
}

// Case 2：max、min 都在右 → good ⟺ l = r − (rMax[r] − rMin[r])
long long maxRminR(int lo, int mid, int hi) {
    long long good = 0;
    for (int r = mid + 1; r <= hi; ++r) {
        int l = r - (rMax[r] - rMin[r]);
        if (l >= lo && l <= mid && lMax[l] < rMax[r] && lMin[l] > rMin[r]) ++good;
    }
    return good;
}

// Case 3：max 在左、min 在右 → good ⟺ l + lMax[l] == r + rMin[r]
// maxPtr 維護 rMax[r] < lMax[l] 的前綴 (mid, maxPtr]；minPtr 維護 rMin[r] < lMin[l] 的後綴 [minPtr, hi]
long long maxLminR(int lo, int mid, int hi) {
    long long good = 0;
    int maxPtr = mid, minPtr = mid + 1;
    for (int l = mid; l >= lo; --l) {
        while (maxPtr + 1 <= hi && rMax[maxPtr + 1] < lMax[l]) {
            ++maxPtr;
            if (maxPtr >= minPtr) ++bucket[maxPtr + rMin[maxPtr]];
        }
        while (minPtr <= hi && rMin[minPtr] >= lMin[l]) {
            if (minPtr <= maxPtr) --bucket[minPtr + rMin[minPtr]];
            ++minPtr;
        }
        good += bucket[l + lMax[l]];
    }
    for (int r = minPtr; r <= maxPtr; ++r) bucket[r + rMin[r]] = 0;
    return good;
}

// Case 4：max 在右、min 在左（與 case 3 對稱）→ good ⟺ rMax[r] − r == lMin[l] − l
// key 加 n 以避免負索引
long long maxRminL(int lo, int mid, int hi) {
    long long good = 0;
    int maxPtr = mid + 1, minPtr = mid;
    for (int l = mid; l >= lo; --l) {
        while (minPtr + 1 <= hi && rMin[minPtr + 1] > lMin[l]) {
            ++minPtr;
            if (minPtr >= maxPtr) ++bucket[rMax[minPtr] - minPtr + n];
        }
        while (maxPtr <= hi && rMax[maxPtr] <= lMax[l]) {
            if (maxPtr <= minPtr) --bucket[rMax[maxPtr] - maxPtr + n];
            ++maxPtr;
        }
        good += bucket[lMin[l] - l + n];
    }
    for (int r = maxPtr; r <= minPtr; ++r) bucket[rMax[r] - r + n] = 0;
    return good;
}

// 所有 l ≤ mid < r 的 good (l, r)
long long countCrossing(int lo, int mid, int hi) {
    buildPrefix(lo, mid, hi);
    return maxLminL(lo, mid, hi) + maxRminR(lo, mid, hi)   // 同側
         + maxLminR(lo, mid, hi) + maxRminL(lo, mid, hi);  // 異側
}

long long solve(int lo, int hi) {
    if (lo == hi) return 1;          // 單一元素必 good
    int mid = (lo + hi) / 2;
    return solve(lo, mid) + solve(mid + 1, hi) + countCrossing(lo, mid, hi);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    while (t--) {
        cin >> n;
        for (int i = 1; i <= n; ++i) cin >> a[i];
        cout << solve(1, n) << '\n';
    }
    return 0;
}

複雜度分析// Complexity Derivation

把 countCrossing(lo, mid, hi) 拆成 5 段（設 K = hi − lo + 1）：

步驟內容單次成本
buildPrefix：lMax/lMin/rMax/rMin掃過左右兩半各一次O(K)
maxLminL（Case 1）每個 l 一次 O(1) 檢查O(K)
maxRminR（Case 2）每個 r 一次 O(1) 檢查O(K)
maxLminR（Case 3, two-pointer）l 走一遍 + maxPtr/minPtr 各走一遍O(K)
maxRminL（Case 4, two-pointer）l 走一遍 + maxPtr/minPtr 各走一遍O(K)
bucket 清理active set 大小 ≤ K，O(K) 歸零O(K)
T(N) = 2 · T(N/2) + O(N)
Master Theorem case 2：a = 2, b = 2, f(N) = O(N) = O(Nlogb a) ⇒ T(N) = Θ(N log N)
項目數量備註
Time (單組)O(N log N)N = 5 × 10⁵ ⇒ 約 10⁷ 次運算
Time (全部 T 組)O(S log S)S ≤ 5 × 10⁵ ⇒ ~10⁷ 級，安全過時限
SpaceO(N)a, lMax, lMin, rMax, rMin 各一支 + bucket 約 2N
Recursion DepthO(log N)N = 5 × 10⁵ ⇒ 約 20 層
Constant factor中四個 case + 兩個 two-pointer + cnt 加減
03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
答案型別 long long n = 5 × 10⁵ 全為「已排序」時答案 = n(n+1)/2 ≈ 1.25 × 10¹¹，遠超 int 上限（2.1 × 10⁹）。 solve、countCrossing 與四個 case 函式都回傳 long long。
跨 test case 殘留 bucket[] 在每個 case 3/4 結束都必須歸零；maxLminR/maxRminL 各自在結尾 局部清掉 active set 範圍 [minPtr, maxPtr]（或對稱）內的 index，避免後續 case 或測資看到髒資料。
bucket 陣列範圍 case 3 的 key 是 r + rMin[r] ∈ [2, 2n]； case 4 的 key 加了 + n 偏移後 ∈ [1, 2n − 1]。 開到 2 · MAXN 才安全。
1-indexed vs 0-indexed 實作採 1-indexed（a[1..n]）。lo == hi 即 base case；mid = (lo + hi)/2； crossing 範圍 L ∈ [lo, mid]、R ∈ [mid+1, hi]。
嚴格 vs 非嚴格 permutation ⇒ 元素皆不同，所以 lMax[l] ≠ rMax[r]（不同位置的 max 不可能相等）， 可放心用 嚴格不等 > / <。
two-pointer 不要 reset case 3/4 內 maxPtr/minPtr 全程單調右移，整個 l 迴圈共用一對指針，不在每個 l 重新初始化。 這是達到 O(N) 而非 O(N²) 的關鍵。
IO 速度 t、n 都到 5 × 10⁵ 級，必加 ios::sync_with_stdio(false); cin.tie(nullptr); 或改用 scanf/printf。 輸出用 '\n' 別用 endl。
04

細節走讀// 4-Case Walkthrough

① maxLminL（CASE 1）：max + min 都在左半

// maxLminL — max+min on LEFT
條件：max = lMax[l]、min = lMin[l]
goodness：lMax[l] − lMin[l] = r − l
       ⇒ r = l + lMax[l] − lMin[l]   ← 對每個 l 直接算出唯一候選 r

驗證右側不破壞：rMax[r] < lMax[l]、rMin[r] > lMin[l]
   （若右側出現更大值或更小值，max/min 就不是「來自左半」了）

for l in [lo, mid]:
    r = l + lMax[l] − lMin[l]
    if r in (mid, hi] AND rMax[r] < lMax[l] AND rMin[r] > lMin[l]:
        good += 1

② maxRminR（CASE 2）：max + min 都在右半 · 對偶

// maxRminR — max+min on RIGHT
條件：max = rMax[r]、min = rMin[r]
goodness：rMax[r] − rMin[r] = r − l
       ⇒ l = r − (rMax[r] − rMin[r])   ← 對每個 r 直接算出唯一候選 l

驗證左側不破壞：lMax[l] < rMax[r]、lMin[l] > rMin[r]

for r in (mid, hi]:
    l = r − (rMax[r] − rMin[r])
    if l in [lo, mid] AND lMax[l] < rMax[r] AND lMin[l] > rMin[r]:
        good += 1

③ maxLminR（CASE 3）：max 在左、min 在右 · two-pointer 計數

// maxLminR — max LEFT, min RIGHT
條件：lMax[l] > rMax[r] AND rMin[r] < lMin[l]
goodness：lMax[l] − rMin[r] = r − l
       ⇒ l + lMax[l] = r + rMin[r]   ← 等式可用 bucket[] hashing

l 由 mid 向 lo 走：lMax[l] 不減、lMin[l] 不增
maxPtr (max-side 滿足者上界) ↗ 只右移：lMax 變大 ⇒ 更多 r 滿足
minPtr (min-side 不滿足者上界+1) ↗ 只右移：lMin 變小 ⇒ 更少 r 滿足
active set = r ∈ [minPtr, maxPtr]

每加入 r ⇒ bucket[r + rMin[r]]++
每移除 r ⇒ bucket[r + rMin[r]]−−
查詢      ⇒ good += bucket[l + lMax[l]]

④ maxRminL（CASE 4）：max 在右、min 在左 · 對偶 two-pointer

// maxRminL — max RIGHT, min LEFT
條件：rMax[r] > lMax[l] AND rMin[r] > lMin[l]
goodness：rMax[r] − lMin[l] = r − l
       ⇒ rMax[r] − r = lMin[l] − l   ← 兩邊都跟「位置」綁在一起

bucket 用 (rMax[r] − r + n) 當 index（避免負數）

l 由 mid 向 lo 走：lMin[l] 不增 ⇒ minPtr 右移（min-side 多收 r）
              lMax[l] 不減 ⇒ maxPtr 右移（max-side 失去 r）
active set = r ∈ [maxPtr, minPtr]

查詢 ⇒ good += bucket[lMin[l] − l + n]

為什麼是 O(N log N)？case 3/4 的 two-pointer 在 l 迴圈結束時，maxPtr 最多走過 hi−mid 步、 minPtr 也最多走過 hi−mid 步 ⇒ 整個 case 在 O(hi − lo + 1) 內完成。每層遞迴掃 O(N)，共 log N 層。
