01

完整實作// Full C++

互動式題目：實作 solve(N)，由 judge 提供 Report(int)。不需 main()。

p117.cpp C++17 
#include <bits/stdc++.h>
#include "lib0117.h"
using namespace std;

const int N_MAX = 1100;
int M[N_MAX][N_MAX];   // 1-indexed N x N 矩陣

// 在 M[rowOff+1..rowOff+N][colOff+1..colOff+N] 內填入 A(N)，
// 值域 {base+1 .. base+2N-1}。base 通常為 0，遞迴中也保持 0。
void build(int N, int rowOff, int colOff, int base) {
    if (N == 1) {
        M[rowOff + 1][colOff + 1] = base + 1;
        return;
    }
    int K = N / 2;
    int shift = 2 * K - 1;             // = N - 1

    // TL：A(K) 原值，值域 {base+1 .. base+2K-1}
    build(K, rowOff,     colOff,     base);
    // BR：同樣 A(K) 原值（與 TL 一樣，不平移）
    build(K, rowOff + K, colOff + K, base);

    // 掃 TL，由 TL 推導 BL 與 TR
    for (int i = 1; i <= K; ++i) {
        for (int j = 1; j <= K; ++j) {
            int tlVal = M[rowOff + i][colOff + j];
            // BL：所有位置 +shift
            M[rowOff + K + i][colOff + j] = tlVal + shift;
            // TR：對角 i==j 填 2N-1；其餘 +shift
            if (i == j) {
                M[rowOff + i][colOff + K + j] = base + 2 * N - 1;
            } else {
                M[rowOff + i][colOff + K + j] = tlVal + shift;
            }
        }
    }
}

void solve(int N) {
    build(N, 0, 0, 0);

    // row-major 輸出
    for (int i = 1; i <= N; ++i) {
        for (int j = 1; j <= N; ++j) {
            Report(M[i][j]);
        }
    }
}

複雜度分析// Complexity Derivation

把 build(N) 一層的工作量拆開：

步驟內容單呼叫成本
遞迴填 TLbuild(K, ...)T(N/2)
遞迴填 BRbuild(K, ...)（另一個 K × K 子塊）T(N/2)
掃 TL 推 BL + TRK × K 個格子，每格寫 BL 1 次、寫 TR 1 次Θ(2K²) = Θ(N²/2)

遞迴式：

T(N) = 2 · T(N/2) + Θ(N²)  ⇒  T(N) = Θ(N²)
Master Theorem case 3：f(N) = Θ(N²) 主宰 N^(log₂ 2) = N ⇒ T(N) = Θ(f(N)) = Θ(N²)。 更直觀：每格只寫一次 ⇒ 總工作量 = 格數 = N²。
項目數量備註
TimeΘ(N²)建構 + 輸出皆 Θ(N²)，同階
SpaceΘ(N²)int M[1100][1100] ≈ 4.8 MB（int = 4B × 1100² ≈ 4.8M）
Recursion DepthO(log N)N = 1024 ⇒ 10 層
Report CallsN²N = 1024 ⇒ 約 10⁶ 次

子任務對照： N ≤ 8 ⇒ 64 個格、64 次 Report； N ≤ 64 ⇒ 4096； N ≤ 1024 滿分 ⇒ 約 10⁶ — 全部都遠在時限內，瓶頸不是時間而是「正確構造」。

03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
必須 #include "lib0117.h" judge 提供的標頭 — Report 的宣告在裡面。沒 include 會編譯失敗：'Report' was not declared in this scope。和 P116 同樣模式。
N 必須是 2 的次方 遞迴每層切半（K = N/2）。題目滿分子任務都保證 N = 2^K，所以不需特判 N 為奇。若被丟非 2 次方的 N 進來，K = N/2 變不對稱（floor 後 2K ≠ N），構造會壞掉。
Report 順序必須 row-major judge 期待 (1,1), (1,2), ..., (1,N), (2,1), ..., (N,N) 的順序 — 不是 col-major、也不是遞迴順序。solve() 末尾的 for i for j Report(M[i][j]) 就是這個順序。
不用 ios::sync_with_stdio 互動題只透過 Report 與 judge 溝通，沒有 cin/cout ⇒ 加 sync_with_stdio 也沒用，刪掉更乾淨。和 P116 同樣模式。
全域矩陣 ≈ 4.8 MB int M[1100][1100] = 4 × 1100² B ≈ 4.6 MB；放函數內會爆 stack，必須全域。N_MAX = 1100 比 1024 略大留餘量。
BR 不能加 shift 很容易直覺地把四塊都「平移到不同值域」，但 BR 必須與 TL 同值。若 BR 加 shift，row N ∪ col N 會多 BR 高值、少 TL 低值，不再覆蓋 {1..2K−1} ⇒ 構造錯。
TR 必須是「對角 = 2N−1，其餘 = +shift」 若 TR 整塊都填 +shift，2N−1 就不會出現 ⇒ row 與 col 的聯集少掉 2N−1。對角線是這個構造的「靈魂」。
base 參數保持 0 遞迴呼叫 build(K, ...) 時，TL 與 BR 都傳 base（同值），不傳 base + shift。BL 與 TR 的 +shift 是在外層直接加到 TL 值上，不是遞迴內部處理的。

細節走讀// Section Walkthrough

① FILL TL

// FILL TL
build(K, rowOff, colOff, base):
    # 遞迴填 K x K 子塊，左上角 = (rowOff+1, colOff+1)
    # 值域 {base+1 .. base+2K-1}

→ 對於 N = 2，K = 1：
   build(1, 0, 0, 0) ⇒ M[1][1] = 1
→ 對於 N = 4，K = 2：
   build(2, 0, 0, 0) ⇒ M[1..2][1..2] = [[1, 3], [2, 1]]

② FILL BR

// FILL BR
build(K, rowOff + K, colOff + K, base):
    # 與 TL 同值（同樣的 base）
    # 不平移、不對角線特別處理

→ 對於 N = 2：M[2][2] = 1
→ 對於 N = 4：M[3..4][3..4] = [[1, 3], [2, 1]]（與 TL 同）

③ FILL BL (+ shift)

// FILL BL
for i in [1, K], j in [1, K]:
    tlVal = M[rowOff + i][colOff + j]
    M[rowOff + K + i][colOff + j] = tlVal + shift

→ 對於 N = 2，shift = 1：
   M[2][1] = M[1][1] + 1 = 2
→ 對於 N = 4，shift = 3：
   M[3..4][1..2] = TL + 3 = [[4, 6], [5, 4]]

④ FILL TR (with diagonal = 2N − 1)

// FILL TR (with diagonal = 2N - 1)
for i in [1, K], j in [1, K]:
    if i == j:
        M[rowOff + i][colOff + K + j] = base + 2N - 1
    else:
        M[rowOff + i][colOff + K + j] = tlVal + shift

→ 對於 N = 2，2N - 1 = 3：
   i = j = 1 ⇒ M[1][2] = 3
→ 對於 N = 4，2N - 1 = 7，shift = 3：
   M[1][3] = 7 (i=j=1, 對角)
   M[1][4] = M[1][2] + 3 = 3 + 3 = 6
   M[2][3] = M[2][1] + 3 = 2 + 3 = 5
   M[2][4] = 7 (i=j=2, 對角)
   ⇒ M[1..2][3..4] = [[7, 6], [5, 7]]

⑤ REPORT ROW-MAJOR

// REPORT ROW-MAJOR
for i in [1, N]:
    for j in [1, N]:
        Report(M[i][j])

→ N = 2 輸出順序：1, 3, 2, 1
→ N = 4 輸出順序：1, 3, 7, 6, 2, 1, 5, 7, 4, 6, 1, 3, 5, 4, 2, 1
