p115.cpp C++17 
#include <iostream>
#include <vector>
using namespace std;

const long long MOD = 10000019;

long long inv_sum = 0;
vector<long long> buf;          // merge 用的暫存陣列

// 合併 a[left..mid] 和 a[mid+1..right]，
// 同時把這次跨界產生的逆序對「兩數和」累進 inv_sum。
void merge(vector<long long>& a, int left, int mid, int right) {
    // remaining = a[left..mid] 之和（mod MOD），左半游標前進時遞減
    long long remaining = 0;
    for (int p = left; p <= mid; ++p) {
        remaining = (remaining + a[p] % MOD) % MOD;
    }

    int i = left;          // 左半游標
    int j = mid + 1;       // 右半游標
    int k = left;          // 寫入暫存陣列的位置

    while (i <= mid && j <= right) {
        if (a[i] <= a[j]) {
            // 取左半，無新逆序對；remaining 遞減扣掉 a[i]
            buf[k++] = a[i];
            remaining = (remaining - a[i] % MOD + MOD) % MOD;
            ++i;
        } else {
            // a[i] > a[j]：左半剩餘 a[i..mid] 全部 > a[j]，
            // 都跟 a[j] 形成逆序對。貢獻 = remaining + a[j] × count
            long long count = mid - i + 1;
            long long contrib = (remaining + (a[j] % MOD) * (count % MOD)) % MOD;
            inv_sum = (inv_sum + contrib) % MOD;
            buf[k++] = a[j];
            ++j;
        }
    }
    while (i <= mid)   buf[k++] = a[i++];
    while (j <= right) buf[k++] = a[j++];

    for (int p = left; p <= right; ++p) a[p] = buf[p];
}

void merge_sort(vector<long long>& a, int left, int right) {
    if (left >= right) return;
    int mid = (left + right) / 2;
    merge_sort(a, left, mid);
    merge_sort(a, mid + 1, right);
    merge(a, left, mid, right);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<long long> a(n);
    for (int i = 0; i < n; ++i) cin >> a[i];

    buf.assign(n, 0);
    merge_sort(a, 0, n - 1);

    cout << inv_sum << '\n';
    return 0;
}

逐段走讀

核心是 merge(a, left, mid, right)：合併兩個已排序子陣列，同時累計跨界逆序對的「兩數和」。

long long remaining = 0;
for (int p = left; p <= mid; ++p) {
    remaining = (remaining + a[p] % MOD) % MOD;
}
初始化 remaining。 用 remaining 維護「左半剩餘元素之和 (mod MOD)」。 merge 開始前一次性算完 a[left..mid] 的總和；之後左半游標前進時遞減，避免每次重新求和。 
if (a[i] <= a[j]) {
    buf[k++] = a[i];
    remaining = (remaining - a[i] % MOD + MOD) % MOD;
    ++i;
}
取左半元素。 比較用原始值，所以即使陣列含大數也直接拿來比。 取走 a[i] 後 remaining 必須扣掉 a[i] % MOD，並加上 MOD 防止負數。 
} else {
    long long count = mid - i + 1;
    long long contrib = (remaining
        + (a[j] % MOD) * (count % MOD)) % MOD;
    inv_sum = (inv_sum + contrib) % MOD;
    buf[k++] = a[j];
    ++j;
}
取右半元素 + 累計貢獻。 a[i] > a[j] 時，左半剩 count = mid - i + 1 個元素，每個都跟 a[j] 構成逆序對。 本步貢獻 = remaining + a[j] × count； (a[j] % MOD) × (count % MOD) ≤ 10⁷ × 10⁶ ≈ 10¹³，仍在 long long 範圍。 
while (i <= mid)   buf[k++] = a[i++];
while (j <= right) buf[k++] = a[j++];
把剩餘的單邊倒進 buf。 若主迴圈結束時左半還沒清空，剩下的元素都 ≤ 右半已被吃光的所有元素 — 不會再產生跨界逆序對，不需更新 inv_sum。 右半同理。 
for (int p = left; p <= right; ++p) a[p] = buf[p];
寫回原陣列。 merge 完成後 a[left..right] 已排序好，回傳給上層遞迴繼續往上 merge。 因為 buf 是全域大小 N 的暫存，所以避免每次 merge 都 new/delete。
// Common Pitfalls
  • 排序比較永遠用 原始值，不要先 mod。陣列裡的值維持原本大小。
  • 累加進 inv_sum 時務必每步都 mod，否則最壞情況 (N=10⁶ 全降序) 會溢位 long long。
  • 減法 (remaining - a[i] % MOD + MOD) % MOD 一定要 + MOD，避免結果為負。
  • buf 用全域而不是每次 merge 內 new：避免 10⁶ × log(10⁶) 次的 malloc / free 開銷。
  • cin / cout 速度：ios::sync_with_stdio(false); cin.tie(nullptr); 必加，否則 10⁶ 筆輸入可能 TLE。
  • 本題 cout << '\n'，避免 endl 的 flush 開銷。
