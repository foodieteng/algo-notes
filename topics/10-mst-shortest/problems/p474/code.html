01

完整實作// Full C++

p474.cpp C++17 
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

// ── 並查集（路徑壓縮 + 按秩合併）
int par[200005], rnk[200005];

int find(int x) {
    while (par[x] != x) { par[x] = par[par[x]]; x = par[x]; }
    return x;
}

bool unite(int a, int b) {
    a = find(a); b = find(b);
    if (a == b) return false;
    if (rnk[a] < rnk[b]) swap(a, b);
    par[b] = a;
    if (rnk[a] == rnk[b]) rnk[a]++;
    return true;
}

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    // 虛擬油源節點 0：國家 i 開井 = 邊 (0, i) 花 cᵢ
    vector<array<ll, 3>> es;
    es.reserve(N + M);
    for (int i = 1; i <= N; i++) {
        ll c; cin >> c;
        es.push_back({c, 0, (ll)i});     // 開井虛邊 (0, i, cᵢ)
    }
    for (int i = 0; i < M; i++) {
        ll u, v, w; cin >> u >> v >> w;
        es.push_back({w, u, v});          // 道路（重邊、自環都無妨）
    }
    sort(es.begin(), es.end());

    for (int i = 0; i <= N; i++) { par[i] = i; rnk[i] = 0; }

    // Kruskal：N+1 個節點連成生成樹需 N 條邊
    ll total = 0; int used = 0;
    for (auto& e : es) {
        ll w = e[0]; int u = e[1], v = e[2];
        if (unite(u, v)) {
            total += w;
            if (++used == N) break;           // 已連好，提早結束
        }
    }

    cout << total << '\n';
}

int main() { solve(); return 0; }

整題精華就是那一行 es.push_back({c, 0, i}) —— 把「開井」翻譯成「連到虛擬源點 0」， 接下來就是教科書 Kruskal。自環（u=v）與重邊都被 find(u)==find(v) 自然略過。

複雜度分析// Complexity Derivation

邊集合大小 = N（虛邊）+ M（道路）：

步驟內容成本
建虛邊每國一條 (0, i, cᵢ)O(N)
讀道路M 條 (u, v, w)O(M)
排序N+M 條邊按權重O((N+M) log)
掃邊 + DSU每邊 find/union 各一次O((N+M) α)
T = O((N + M) log(N + M))
排序主導；DSU 的 α 近乎常數
項目數量備註
TimeO((N+M) log)N=M=2×10⁵ ⇒ 實測約 1 秒
SpaceO(N + M)es 陣列 + par/rnk（大小 N+1）
數值型別long long權和達 2×10¹⁴；c, w ≤ 10⁹
生成樹邊數N節點 0~N 共 N+1 個 ⇒ N 條邊
03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
虛擬源點 0 並查集大小要開 N+1（含節點 0），初始化 par[0..N]。漏掉 0 會 RE 或答案錯。
停在 N 條邊 節點是 0~N 共 N+1 個 ⇒ 生成樹要 N 條邊（不是 N−1）。used == N 才 break。
權和開 long long 最多 N 條邊、每條 ≤ 10⁹ ⇒ 權和達 2×10¹⁴。total 與排序鍵 w 都用 ll。
自環 / 重邊 道路可能 u=v（自環）或重複連同一對。Kruskal 用 find(u)==find(v) 自然略過，不必特判。
圖一定能完成 因為每國都有「開井」虛邊到 0 ⇒ 圖必連通、MST 一定湊得滿 N 條，不會無解。
I/O 速度 N+M 達 4×10⁵ ⇒ sync_with_stdio(false); cin.tie(nullptr); 必加。

細節走讀// Section Walkthrough

① 建虛邊（開井）

// well edges
for i in [1, N]:
    es.push_back({c_i, 0, i})    ← 開井 = 連虛源 0，權重 c_i

② 放道路

// road edges
for each road (u, v, w):
    es.push_back({w, u, v})      ← 重邊、自環照丟，不特判

③ 排序 + 初始化 DSU(0..N)

// sort & init
sort(es)                          ← array 字典序先比 w
for i in [0, N]: par[i] = i       ← 注意含節點 0

④ Kruskal 取 N 條

// kruskal
total = 0; used = 0
for (w, u, v) in es:
    if unite(u, v):
        total += w
        if ++used == N: break     ← N+1 點 ⇒ N 條邊
