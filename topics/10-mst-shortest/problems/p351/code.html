慘字道路規劃

01

完整實作// Full C++

p351.cpp C++17 
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll INF = (ll)1e18;

int N;
ll dist[505][505];   // 目前最短距離（只用已加入城市當中繼）
ll c[505][505];      // 原始鄰接矩陣

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> N;
    for (int i = 1; i <= N; i++)
        for (int j = 1; j <= N; j++) {
            ll x; cin >> x;
            c[i][j] = (x == -1 ? INF : x);   // -1 ⇒ 沒有直接道路
        }

    for (int i = 1; i <= N; i++)
        for (int j = 1; j <= N; j++)
            dist[i][j] = INF;

    for (int k = 1; k <= N; k++) {
        // 1) 加入城市 k：填入 k 與 {1..k} 之間的直接邊
        for (int i = 1; i <= k; i++) {
            dist[i][k] = min(dist[i][k], c[i][k]);
            dist[k][i] = min(dist[k][i], c[k][i]);
        }
        dist[k][k] = min(dist[k][k], (ll)0);

        // 2) 先用「之前的城市 m」當中繼，修好 k 的行/列（多跳進出 k）
        for (int m = 1; m < k; m++)
            for (int i = 1; i <= k; i++) {
                if (dist[i][m] < INF && dist[m][k] < INF)
                    dist[i][k] = min(dist[i][k], dist[i][m] + dist[m][k]);
                if (dist[k][m] < INF && dist[m][i] < INF)
                    dist[k][i] = min(dist[k][i], dist[k][m] + dist[m][i]);
            }

        // 3) 以 k 為中繼，鬆弛 {1..k} 之間所有對
        for (int i = 1; i <= k; i++)
            for (int j = 1; j <= k; j++)
                if (dist[i][k] < INF && dist[k][j] < INF)
                    dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);

        // 4) 慘字度 = max over i,j in {1..k} of dist[i][j]（有 ∞ ⇒ -1）
        ll worst = 0; bool bad = false;
        for (int i = 1; i <= k && !bad; i++)
            for (int j = 1; j <= k; j++) {
                if (dist[i][j] >= INF) { bad = true; break; }
                worst = max(worst, dist[i][j]);
            }

        cout << (bad ? -1 : worst);
        cout << (k == N ? '\n' : ' ');
    }
}

int main() { solve(); return 0; }

步驟 2 是這題的關鍵：少了它，k 的進出最短路只算到「直接邊」⇒ 漏掉多跳捷徑、答案偏大。

複雜度分析// Complexity Derivation

第 k 層的工作量：

步驟內容單層成本
填直接邊k 的行/列O(k)
修 k 行列m × i 兩層迴圈O(k²)
k 當中繼鬆弛i × j 兩層O(k²)
掃 maxi × j 兩層O(k²)
Σk=1..N O(k²) = O(N³)
每層 O(k²)，加總即 O(N³)，跟跑一次完整 Floyd 同量級
項目數量備註
TimeO(N³)N=500 ⇒ ~1.25×10⁸，實測 0.3 秒
SpaceO(N²)dist + c 兩個 505×505 矩陣
數值型別long long距離其實 ≤ 5×10⁵，但 INF 相加要防溢位，索性全用 ll
Constant factor中每層三趟 O(k²)，常數約 3
03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
少了「修 k 行列」 只填 k 的直接邊、直接以 k 鬆弛 ⇒ 進出 k 的多跳捷徑沒算到。樣本最後一項會錯成 12（正解 11）。步驟 2 不可省。
−1 要轉成 INF c[i][j] = -1 是「沒有直接道路」，要轉成 INF；別當成「距離 −1」拿去做 min。
不可達輸出 −1 掃 max 時若有任何 dist[i][j] = INF，代表這個前綴有點互相到不了 ⇒ 整個輸出 -1（大慘字規劃）。
INF 相加防溢位 dist[i][k] + dist[k][j] 前先判兩者 < INF，否則 INF+INF 會溢位變垃圾甚至負數。
只掃 {1..k} 第 k 層只考慮「已加入的城市」，max 與鬆弛都只跑 1..k，別掃到還沒加入的 k+1..N。
輸出格式 N 個整數一行、空白隔開，最後接換行；別忘了 dist[i][i]=0 不影響 max。

細節走讀// Section Walkthrough

① 讀矩陣（−1 → INF）

// read
c[i][j] = (x == -1) ? INF : x    ← -1 代表沒有直接道路
dist[][] 全部初始化成 INF

② 加入城市 k：填直接邊

// step 1
for i in [1, k]:
    dist[i][k] = min(dist[i][k], c[i][k])   ← 進 k 的直接邊
    dist[k][i] = min(dist[k][i], c[k][i])   ← 出 k 的直接邊
dist[k][k] = 0

③ 修 k 的行列（多跳進出）

// step 2 (關鍵)
for m in [1, k-1]:           ← 用之前的城市當中繼
    for i in [1, k]:
        dist[i][k] = min(dist[i][k], dist[i][m] + dist[m][k])
        dist[k][i] = min(dist[k][i], dist[k][m] + dist[m][i])

④ 以 k 鬆弛全對 + 掃 max

// step 3 + 4
for i in [1,k]: for j in [1,k]:
    dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j])
worst = max dist[i][j]；若有 INF ⇒ 輸出 -1
