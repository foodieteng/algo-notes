完整實作// Full C++

p507.cpp C++17 
#include <bits/stdc++.h>
using namespace std;

struct Point {
    long long x, y;                                 // 座標 ≤ 10⁹ ⇒ 用 long long 防 dx² 溢位
};

// 距離平方：純整數，避開 sqrt 浮點誤差
long long dist2(const Point& p1, const Point& p2) {
    long long dx = p1.x - p2.x;
    long long dy = p1.y - p2.y;
    return dx * dx + dy * dy;
}

// 回傳 pts[lo..hi] 內最近兩點的距離平方
long long closest(vector<Point>& pts, int lo, int hi) {
    int n = hi - lo + 1;

    // ── BASE: ≤ 3 點直接 brute（最多 C(3,2) = 3 對）
    if (n <= 3) {
        long long d = LLONG_MAX;

        for (int i = lo; i <= hi; i++) {
            for (int j = i + 1; j <= hi; j++) {
                d = min(d, dist2(pts[i], pts[j]));
            }
        }

        return d;
    }

    // ── 切半（pts 已按 x 排序，中位數即 pts[mid].x）
    int mid = lo + (hi - lo) / 2;                   // overflow-safe
    long long midX = pts[mid].x;

    // ── 左右各遞迴，取較小者當門檻 d
    long long dL = closest(pts, lo, mid);
    long long dR = closest(pts, mid + 1, hi);
    long long d = min(dL, dR);

    // ── STRIP: 只收 |x − midX| < √d 的點 (其餘不可能成跨界對)
    vector<Point> strip;

    for (int i = lo; i <= hi; i++) {
        long long dx = pts[i].x - midX;
        if (dx * dx >= d) continue;                 // 平方比，免 sqrt

        strip.push_back(pts[i]);
    }

    // ── strip 按 y 排序，每點只比後 7 個（pigeonhole 上界）
    sort(strip.begin(), strip.end(), [](const Point& p1, const Point& p2) {
        return p1.y < p2.y;
    });

    for (int i = 0; i < (int)strip.size(); i++) {
        for (int j = i + 1; j < (int)strip.size() && j - i < 8; j++) {
            long long dy = strip[j].y - strip[i].y;
            if (dy * dy >= d) break;                // y 已超過 √d ⇒ 後面也都超，提早終止

            d = min(d, dist2(strip[i], strip[j]));
        }
    }

    return d;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<Point> pts(n);

    for (int i = 0; i < n; i++) {
        cin >> pts[i].x >> pts[i].y;
    }

    // 先按 x 排序一次（x 同就比 y）— 整個遞迴沿用
    sort(pts.begin(), pts.end(), [](const Point& p1, const Point& p2) {
        if (p1.x != p2.x) return p1.x < p2.x;
        return p1.y < p2.y;
    });

    cout << closest(pts, 0, n - 1) << '\n';

    return 0;
}

複雜度分析// Complexity Derivation

把 closest(lo, hi) 一層的工作量分成 3 段：

步驟內容單層成本
遞迴左右兩個 N/2 子問題2 · T(N/2)
收集 strip掃過 N 個點檢查 |x − midX|² < dO(N)
strip 排序sort by y 於 strip 內O(N log N)
strip 比對每點看至多 7 個鄰居（pigeonhole）O(N)
T(N) = 2 · T(N/2) + O(N log N)
展開：合併工作每層 N log N，共 log N 層 ⇒ T(N) = O(N log² N)
項目數量備註
TimeO(N log² N)每層 strip 排序貢獻一個 log
SpaceO(N)strip 暫存陣列；遞迴堆疊 O(log N)
Recursion DepthO(log N)N = 2 × 10⁵ ⇒ 約 18 層
Constant factor低整數比較、無浮點、無 sqrt

若把 strip 排序也做進主遞迴 (像 merge sort 那樣邊遞迴邊維護 y 排序的副本)， 可消掉一個 log 變 O(N log N)。但 code 變複雜，本問題 N = 2 × 10⁵ 不需要那麼極致。

03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
long long 全程 座標 ≤ 10⁹，Δx² 可達 4 × 10¹⁸ — 逼近 long long 上限 (9.2 × 10¹⁸)。int 在乘法當下就溢位。
strip 條件用平方 dx * dx < d 等價於 |dx| < √d，避開 sqrt 的浮點誤差。整題純整數。
base case 設 n ≤ 3 n = 1 或 2 時 mid 落在唯一點上；設 ≤ 3 邏輯簡單也夠快（最多 3 對 brute）。
strip 提前終止 內層加 if (dy² ≥ d) break; — y 已排序，後面只會更遠。實際大幅縮減常數。
避免重複排序 原 pts 按 x 排序一次後就不再動；strip 每層自己排自己（只含 strip 內的點，不是整批）。
I/O 速度 ios::sync_with_stdio(false); cin.tie(nullptr); 必加；N = 2 × 10⁵ 筆輸入endl 別用，'\n' 即可。

細節走讀// Section Walkthrough

① base case (n ≤ 3)

// n <= 3
double for-loop 比所有 C(n, 2) 對 ⇒ 最多 3 次 dist2() 呼叫。
回傳該子集的最小 d²。沒有遞迴往下走，stack 在這停。

② 切半 & 遞迴

// split
mid  = lo + (hi - lo) / 2  ← overflow-safe 寫法
midX = pts[mid].x          ← 沿用 x 排序好的中位數
dL = closest(lo,    mid)
dR = closest(mid+1, hi)
d  = min(dL, dR)

③ 收集 strip

// strip collect
for i in [lo, hi]:
    dx = pts[i].x - midX
    if dx * dx >= d: continue   ← 太遠 ⇒ 跳過（平方比，純整數）
    strip.push_back(pts[i])

strip 大小最壞 O(N)，但平均遠小於此 — 越往上層 d 越大，strip 越窄反而越快。

④ strip 排序 + 7 鄰居檢查

// strip check
sort(strip by y)
for i = 0..|strip|-1:
    for j = i+1..min(|strip|-1, i+7):
        dy = strip[j].y - strip[i].y
        if dy * dy >= d: break    ← 早結束
        d = min(d, dist2(strip[i], strip[j]))
