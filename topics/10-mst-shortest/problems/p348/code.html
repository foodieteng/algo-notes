01

完整實作// Full C++

p348.cpp C++17 
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

void solve() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;
    while (T--) {
        int N, M, S, E;
        ll F;
        cin >> N >> M >> S >> E >> F;

        // 邊權 = 把整批 F 箱都走這條邊的花費（折扣一次算進去）
        // 凹成本（折扣）⇒ 全部集中在一條路徑最省 ⇒ 退化成最短路
        vector<vector<pair<int, ll>>> adj(N + 1);
        for (int i = 0; i < M; i++) {
            ll A, B, C, D, Cp;
            cin >> A >> B >> C >> D >> Cp;

            ll w;
            if (F <= D) w = F * C;                 // 沒超過 D 箱：全額 C
            else        w = D * C + (F - D) * Cp;  // 超出部份折扣 C'

            adj[A].push_back({(int)B, w});
        }

        // Dijkstra（邊權非負）
        const ll INF = LLONG_MAX;
        vector<ll> dist(N + 1, INF);
        priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<>> pq;
        dist[S] = 0;
        pq.push({0, S});
        while (!pq.empty()) {
            auto [d, u] = pq.top(); pq.pop();
            if (d > dist[u]) continue;           // 過期條目，跳過
            for (auto [v, w] : adj[u]) {
                if (d + w < dist[v]) {
                    dist[v] = d + w;
                    pq.push({dist[v], v});
                }
            }
        }

        cout << dist[E] << '\n';
    }
}

int main() { solve(); return 0; }

整段精華只有那 2 行邊權判斷：把「分段折扣」一次折成單一數字，後面就是教科書 Dijkstra。

複雜度分析// Complexity Derivation

單組測資拆解：

步驟內容成本
讀邊 + 壓權每條邊算一次分段花費O(M)
Dijkstra堆 + 鬆弛 M 條邊O(M log N)
輸出印 dist[E]O(1)
T = O(T · M log N)

項目數量備註
TimeO(T · M log N)每組壓邊權 + 一次 Dijkstra
SpaceO(N + M)鄰接表 + dist 陣列
數值型別long longF·C 最大 ~5×10¹³，路徑和 ~5×10¹⁵
Constant factor低圖極小，堆操作可忽略
03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
別套 MCMF 折扣是凹成本（邊際遞減），最小成本流需要凸成本。硬把邊拆成「D 箱 C + 其餘 C'」兩條，MCMF 會先偷拿便宜的 C' ⇒ 算出比實際更低、錯誤的答案。
邊權是整批不是單價 w 要算「F 箱整批」的花費，不是「每箱單價」。先壓成單一邊權，最短路才有意義。
F ≤ D 的分支 若 F 沒超過 D，整批都是全額 F·C，吃不到折扣。漏判會錯用折扣價。
long long 全程 F·C 達 10⁹×50216 ≈ 5×10¹³；路徑最多 ~100 條邊 ⇒ 總和 ~5×10¹⁵。int 必爆。連 A,B,C,D,C' 讀進來都用 ll 最省心。
有向邊 路線是 A → B 單向（東→西），不可反走。只加單向邊。
多筆測資清空 adj / dist 宣告在 while(T--) 內，每組自動乾淨。
