01

完整實作// Full C++

p604.cpp C++17 
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;

vector<int> adj[MAXN];          // 鄰接表：adj[u] = u 的所有卦源對象
int color[MAXN];                // −1 未著色 / 0 / 1（兩種性別）
int bfsQueue[MAXN];            // 自己的環形佇列，比 std::queue 快

// 從 start 出發 BFS 二著色；回傳 true = 這塊連通塊可二分
bool canColor(int start) {
    int head = 0, tail = 0;
    color[start] = 0;
    bfsQueue[tail++] = start;

    while (head < tail) {
        int u = bfsQueue[head++];

        for (int v : adj[u]) {
            if (color[v] == -1) {           // 還沒塗 → 塗成 u 的反色
                color[v] = color[u] ^ 1;
                bfsQueue[tail++] = v;
            } else if (color[v] == color[u]) {  // 已塗且同色 → 奇環
                return false;
            }
        }
    }

    return true;
}

void solve() {
    int n, m;
    cin >> n >> m;

    for (int i = 0; i < n; i++) {     // 每筆測資都要重置
        adj[i].clear();
        color[i] = -1;
    }

    for (int e = 0; e < m; e++) {
        int a, b;
        cin >> a >> b;
        adj[a].push_back(b);        // 無向邊，兩邊都加
        adj[b].push_back(a);
    }

    bool ok = true;

    for (int i = 0; i < n; i++) {     // 圖可能不連通 → 每個未著色點各跑一次
        if (color[i] == -1 && !canColor(i)) {
            ok = false;
            break;
        }
    }

    cout << (ok ? "NORMAL.\n" : "RAINBOW.\n");
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;
    while (T--) solve();

    return 0;
}

複雜度分析// Complexity Derivation

單筆測資的工作量拆成 3 段：

步驟內容成本
重置清空 n 個 adj[]、color[] 設 −1O(n)
建鄰接表讀 m 條邊，每條 push 兩次O(m)
BFS 著色每點進出 queue 各一次、每邊掃兩次O(n + m)
T = O(n + m)  每筆  ·  總 O(T · (n + m))
T ≤ 10、n ≤ 10⁵、m ≤ 2 × 10⁵ ⇒ 最壞約 3 × 10⁶，秒過
項目數量備註
TimeO(n + m)線性掃完整張圖
SpaceO(n + m)鄰接表 O(m) + color/queue O(n)
BFS 廣度O(n)bfsQueue 最壞裝整個連通塊
Constant factor低自製陣列佇列 + 反色用 XOR，無浮點

用 BFS（陣列佇列）而非遞迴 DFS：n = 10⁵ 的鏈狀圖遞迴會爆 stack，BFS 完全沒這風險。

03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
多筆測資要重置 T 筆共用全域陣列；每筆開頭清空 adj[0..n−1]、color[] 設 −1。漏了會帶髒資料。
圖不一定連通 必須對每個未著色點各跑一次 BFS，不能只從點 0 跑一次。孤立點 / 多個連通塊都算合理。
無向邊兩邊都加 adj[a].push_back(b) 也要 adj[b].push_back(a)，否則漏掉反向就抓不到環。
自環 / 重邊 自環 a == b（自己跟自己）必同色 ⇒ 直接 RAINBOW；本寫法 BFS 取到 v == u 時 color 相同會正確判 false。重邊不影響正確性。
遞迴 vs BFS n = 10⁵ 的長鏈用 DFS 遞迴會 stack overflow；本題用 BFS 陣列佇列避開。
輸出格式 注意題目要求帶句點：NORMAL. / RAINBOW.，別漏掉那個 .。
04

細節走讀// Section Walkthrough

① 每筆測資重置

// reset
for i in [0, n):
    adj[i].clear()      ← 上一筆的邊全清掉
    color[i] = -1       ← -1 代表「未著色」
讀 m 條邊，無向 ⇒ adj[a] 推 b、adj[b] 推 a

② 逐連通塊跑 BFS

// component loop
for i in [0, n):
    if color[i] == -1:        ← 還沒碰過的點 ⇒ 新連通塊
        if not canColor(i): RAINBOW; break
全部塗完沒衝突 ⇒ NORMAL

③ BFS 二著色（核心）

// canColor(start)
color[start] = 0; 推進 queue
while queue 非空:
    u = pop()
    for v in adj[u]:
        if color[v] == -1:           ← 沒塗過
            color[v] = color[u] ^ 1   ← 塗反色（XOR 1：0↔1）
            推進 queue
        elif color[v] == color[u]:   ← 已塗且同色
            return false              ← 撞到奇環！

color[u] ^ 1 是把 0 變 1、1 變 0 的標準寫法 — 每條樹邊兩端必為反色，唯一會「同色相鄰」的就是把奇環封口的那條非樹邊。

④ 為何用自製陣列佇列

// queue
int bfsQueue[MAXN]; int head, tail;
push: bfsQueue[tail++] = x
pop : x = bfsQueue[head++]
每點最多進一次 ⇒ tail 不會超過 n ⇒ 不必環形繞回，直接線性陣列。
比 std::queue 少了配置/釋放節點的常數。
