喵喵抓老鼠

01

完整實作// Full C++

p43.cpp C++17 
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 105;

char grid[MAXN][MAXN];          // 迷宮：'#' 牆 / '.' 路 / 'K' 貓 / '@' 老鼠
int  dist[MAXN][MAXN];          // 到該格的最短步數，-1 = 未訪

// BFS 用的手動 queue（int 陣列，不吃呼叫堆疊）
int qx[MAXN * MAXN];
int qy[MAXN * MAXN];

int dx[4] = {-1, 1, 0, 0};      // 上下左右
int dy[4] = {0, 0, -1, 1};

int N, M;

void solve() {
    cin >> N >> M;

    int startX = -1, startY = -1;

    for (int i = 0; i < N; i++) {
        cin >> grid[i];                        // 整行讀入
        for (int j = 0; j < M; j++) {
            dist[i][j] = -1;                    // 多筆測資：每格重置
            if (grid[i][j] == 'K') { startX = i; startY = j; }
        }
    }

    // ── BFS from 貓 K ──
    int head = 0, tail = 0;
    qx[tail] = startX;
    qy[tail] = startY;
    tail++;
    dist[startX][startY] = 0;

    int ans = -1;                            // -1 = 還沒找到老鼠

    while (head < tail) {
        int x = qx[head];
        int y = qy[head];
        head++;

        if (grid[x][y] == '@') {            // 第一個取出的老鼠 ⇒ 最近
            ans = dist[x][y];
            break;
        }

        for (int d = 0; d < 4; d++) {
            int nx = x + dx[d];
            int ny = y + dy[d];

            if (nx < 0 || nx >= N || ny < 0 || ny >= M) continue;
            if (grid[nx][ny] == '#') continue;        // 牆
            if (dist[nx][ny] != -1) continue;       // 已訪

            dist[nx][ny] = dist[x][y] + 1;      // 入列即定 dist（等同 visited）
            qx[tail] = nx;
            qy[tail] = ny;
            tail++;
        }
    }

    if (ans == -1) cout << "= =\"" << '\n';   // 走不到任何老鼠
    else            cout << ans << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;
    while (T--) solve();

    return 0;
}

注意輸出：走不到時印 = ="（等號、空白、等號、雙引號）。 C++ 字串字面值寫成 "= =\"" — 結尾的雙引號要跳脫成 \"。

複雜度分析// Complexity Derivation

步驟內容成本
讀圖 + 找 K掃過 N×M 格各一次O(N × M)
BFS 入列每格最多入列一次（入列即定 dist）O(N × M)
BFS 出列 + 看四方向每格出列一次，常數 4 鄰居O(4 · N × M)
T = O(N × M) = O(V + E)
網格圖 V = N×M、E ≈ 4V ⇒ BFS 線性
項目數量備註
TimeO(N × M)100×100×31 ≈ 3×10⁵，秒過
SpaceO(N × M)grid + dist + queue 各約 10⁴
遞迴深度0（迭代）BFS 用全域陣列，不爆 stack
Constant factor低整數索引、4 方向 if
03

邊界與陷阱// Pitfalls

陷阱說明 / 解法
用 BFS 不用 DFS DFS 第一次碰到的 @ 不保證最短。最短路一定要 BFS（FIFO 逐層）。
入列即定 dist dist != -1 同時當「已訪」用。若等出列才標，同格會被多個鄰居重複入列 ⇒ queue 爆。
第一個取出的 @ 才回傳 不能「入列時」就回傳 — 入列順序不等於距離順序的終點判定。要在出列（pop）時判斷。
多隻老鼠、一隻貓 @ 可能多個 ⇒ 不能假設只有一個終點；K 唯一 ⇒ 單源即可。
走不到要印 = =" queue 空了 ans 仍 −1 ⇒ 印 = ="。C++ 寫 "= =\""（跳脫尾雙引號）。
多筆要清 dist 每筆讀圖時把 dist 全設 −1。漏清會誤判已訪 ⇒ 答案錯。
K 可踩、@ 可踩 BFS 擴散時 '.'、'@' 都算可走（只有 '#' 擋）；@ 也要能被走進去才算抓到。

細節走讀// Section Walkthrough

① 讀圖 + 找起點 + 清 dist

// read
for i in [0,N): cin >> grid[i]
  for j in [0,M):
    dist[i][j] = -1          ← 多筆測資必清
    if grid[i][j]=='K': (startX,startY)=(i,j)

② BFS 主迴圈 — pop 時判斷終點

// bfs
push K; dist[K]=0
while queue 非空:
    (x,y) = pop front
    if grid[x][y]=='@': ans = dist[x][y]; break   ← 第一個取出的 @ 最近
    for d in 4 方向:
        (nx,ny)=(x+dx,y+dy)
        略過: 出界 / '#' / dist!=-1
        dist[nx][ny]=dist[x][y]+1; push (nx,ny)

head/tail 指標只往前走 ⇒ 每格 push 一次、pop 一次。 dist 沿父鏈 +1 累加，所以 pop 出來的就是真正最短步數。

③ 為什麼判斷要放在「pop」而非「push」

// 反例：若在 push 時就回傳
某個遠的 @ 可能比近的 @ 先被「push」進 queue（看擴散方向）
但它的 dist 更大 → push 順序 ≠ 由近到遠
∴ 一定要等「pop」時才確認：pop 順序 = dist 由小到大
