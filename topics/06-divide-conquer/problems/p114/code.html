p114.cpp C++17 
#include <iostream>
#include "lib0114.h"
using namespace std;

// 鋪滿左上角為 (top, left)、邊長為 n 的子棋盤；
// (blockedX, blockedY) 是已被佔用、不能再放 L 的格子。
void tile(int top, int left, int n, int blockedX, int blockedY) {
    if (n == 2) {
        // 4 格扣掉 blocked 後剩 3 格本身就是 L
        int xs[3], ys[3], k = 0;
        for (int dx = 0; dx < 2; ++dx) {
            for (int dy = 0; dy < 2; ++dy) {
                int x = top + dx, y = left + dy;
                if (x == blockedX && y == blockedY) continue;
                xs[k] = x; ys[k] = y; ++k;
            }
        }
        Report(xs[0], ys[0], xs[1], ys[1], xs[2], ys[2]);
        return;
    }

    const int half = n / 2;
    const int midX = top  + half - 1;   // 上半最後一列
    const int midY = left + half - 1;   // 左半最後一行

    // 4 個象限「最靠近中心」的格子座標 (TL, TR, BL, BR)
    int cx[4] = { midX,     midX,     midX + 1, midX + 1 };
    int cy[4] = { midY,     midY + 1, midY,     midY + 1 };

    // blocked 在哪個象限？  0=TL, 1=TR, 2=BL, 3=BR
    const int bq = (blockedX > midX) * 2 + (blockedY > midY);

    // 中央 L：3 cell = 不含 blocked 的 3 個象限的最角落
    int lx[3], ly[3], k = 0;
    for (int q = 0; q < 4; ++q) {
        if (q == bq) continue;
        lx[k] = cx[q]; ly[k] = cy[q]; ++k;
    }
    Report(lx[0], ly[0], lx[1], ly[1], lx[2], ly[2]);

    // 對 4 象限分別遞迴；blocked 不在的象限以「剛放下的角落格」當新 blocked
    int ox[4] = { top,        top,         top + half, top + half };
    int oy[4] = { left,       left + half, left,       left + half };
    for (int q = 0; q < 4; ++q) {
        int nx = (q == bq) ? blockedX : cx[q];
        int ny = (q == bq) ? blockedY : cy[q];
        tile(ox[q], oy[q], half, nx, ny);
    }
}

void solve(int N, int X, int Y) {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    tile(1, 1, N, X, Y);
}

逐段走讀

核心是 tile(top, left, n, blockedX, blockedY)：對一個左上角 (top, left)、邊長 n、已有一格 (blockedX, blockedY) 不可使用的子板鋪 L。

Base case。 2×2 板共 4 格，扣掉 blocked 後剩 3 格必然構成 L。 直接 double-loop 掃 4 格，跳過 blocked，把剩 3 格塞進 xs / ys 後 Report。 
const int half = n / 2;
const int midX = top  + half - 1;
const int midY = left + half - 1;

int cx[4] = { midX,   midX,   midX+1, midX+1 };
int cy[4] = { midY,   midY+1, midY,   midY+1 };
中央 4 角座標。 每個象限「最靠近中心」的格子，按 TL / TR / BL / BR 順序排成兩個平行陣列： (cx[0], cy[0]) = (midX, midY)，依此類推。 這 4 格就是放置中央 L 的候選。 
const int bq = (blockedX > midX) * 2
             + (blockedY > midY);
用位元把象限編號。 (blockedX > midX) 高位、(blockedY > midY) 低位，組出 0~3：
· TL = 0：blockedX ≤ midX 且 blockedY ≤ midY
· TR = 1：blockedX ≤ midX 且 blockedY > midY
· BL = 2：blockedX > midX 且 blockedY ≤ midY
· BR = 3：blockedX > midX 且 blockedY > midY

    lx[k] = cx[q]; ly[k] = cy[q]; ++k;

Report(lx[0], ly[0],
       lx[1], ly[1],
       lx[2], ly[2]);
放中央 L。 收集「不含 blocked」3 個象限的最角落格塞進 lx / ly，再 Report 出去。 注意這 3 格之後在各自象限遞迴時會被當成 blocked — 它們已經被「占用」掉。 
int ox[4] = { top,  top,         top+half, top+half };
int oy[4] = { left, left+half,   left,     left+half };

for (int q = 0; q < 4; ++q) {
    int nx = (q == bq) ? blockedX : cx[q];
    int ny = (q == bq) ? blockedY : cy[q];
    tile(ox[q], oy[q], half, nx, ny);
}
遞迴 4 個象限。 (ox[q], oy[q]) 是該象限的左上角。 blocked 落在哪個象限，那象限就用原 blocked遞迴； 其他 3 個象限則用剛放下的角落格當 new blocked。 這保證所有 4 個子問題都是「方板 + 1 blocked」的同形式。
// Common Pitfalls
  • 座標 1-indexed：題目給 (X, Y) 從 1 開始，tile 的初始呼叫要傳 tile(1, 1, N, X, Y)。
  • midX = top + half − 1：是上半的最後一列，不是中線；下半從 midX + 1 開始。
  • Base case 不需要再判斷象限 — n=2 時 4 格枚舉即可。
  • 本題不需 cin/cout，ios::sync_with_stdio 純粹習慣放著（若有額外讀檔再用）。
